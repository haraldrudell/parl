    21  func UserHomeDir() (homeDir string) {
    22  	homeDir = getProcessOwnerHomeDir()
    23  	if homeDir == "" {
    24  		var err error
    25  		homeDir, err = os.UserHomeDir() // use $HOME environment variable
    26  		if err != nil {
    27  			panic(error116.Errorf("os.UserHomeDir: '%w'", err))
    28  		}
    29  		if homeDir == "" {
    30  			panic(error116.New("failed to obtain home directory"))
    31  		}
    32  	}
    33  	return
    34  }

    36  // HomeDir creates levels of directories in users’s home
    37  func HomeDir(relPaths string) (dir string) {
    38  	homeDir := UserHomeDir()
    39  	dir = path.Join(homeDir, relPaths)
    40  	if err := os.MkdirAll(dir, 0700); err != nil {
    41  		if !errors.Is(err, os.ErrExist) {
    42  			panic(error116.Errorf("os.MkdirAll: %w", err))
    43  		}
    44  	}
    45  	return
    46  }

    48  func getProcessOwnerHomeDir() (homeDir string) {
    49  	userID := os.Geteuid()
    50  	if userID == -1 { // on Windows, -1 is returned
    51  		return
    52  	}
    53  	userdata, err := user.LookupId(strconv.Itoa(userID))
    54  	if err == nil {
    55  		homeDir = userdata.HomeDir
    56  	}
    57  	return
    58  }

    60  // ShortHostname gets hostname without domain part
    61  func ShortHostname() (host string) {
    62  	var err error
    63  	if host, err = os.Hostname(); err != nil {
    64  		panic(parl.Errorf("os.Hostname: '%w'", err))
    65  	}
    66  	if index := strings.Index(host, "."); index != -1 {
    67  		host = host[:index]
    68  	}
    69  	return
    70  }
