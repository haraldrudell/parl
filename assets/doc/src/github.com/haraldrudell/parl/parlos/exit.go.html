    29  // Exit1OneLine terminate the command echoing to stderr returning status code 1
    30  func Exit1OneLine(err error) {
    31  	if err == nil {
    32  		err = errors.New("Exit1OneLine with err nil")
    33  	}
    34  	fmt.Fprintf(os.Stderr, "%v\n", err)
    35  	OsExit(statusCode)
    36  }

    38  // Exit terminate the command echoing to stderr returning status code
    39  func Exit(code int, err error) {
    40  	if err != nil {
    41  		fmt.Fprintf(os.Stderr, "%+v\n", err)
    42  	}
    43  	OsExit(code)
    44  }
