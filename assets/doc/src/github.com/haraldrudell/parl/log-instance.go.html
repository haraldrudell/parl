    26  // LogInstance provide logging delegating to log.Output
    27  type LogInstance struct {
    28  	isSilence         AtomicBool
    29  	isDebug           AtomicBool
    30  	infoLock          sync.RWMutex
    31  	infoRegexp        *regexp.Regexp // behing infoLock
    32  	output            func(calldepth int, s string) error
    33  	stackFramesToSkip int
    34  }

    36  // NewLog gets a logger for Fatal and Warning for specific output
    37  func NewLog(writers ...io.Writer) (lg *LogInstance) {
    38  	var writer io.Writer
    39  	if len(writers) > 0 {
    40  		writer = writers[0]
    41  	}
    42  	if writer == nil {
    43  		writer = os.Stderr
    44  	}
    45  	logger := logMap[writer]
    46  	if logger == nil {
    47  		logger = log.New(writer, "", 0)
    48  	}
    49  	return &LogInstance{
    50  		output:            logger.Output,
    51  		stackFramesToSkip: logInstDefFrames,
    52  	}
    53  }

    55  // NewLog gets a logger for Fatal and Warning for specific output
    56  func newLog(writer io.Writer, extraStackFramesToSkip int) (lg *LogInstance) {
    57  	if extraStackFramesToSkip < 0 {
    58  		Errorf("newLog extraStackFramesToSkip < 0: %d", extraStackFramesToSkip)
    59  	}
    60  	lg = NewLog(writer)
    61  	lg.stackFramesToSkip += extraStackFramesToSkip
    62  	return
    63  }

    81  // Debug outputs only if debug is configured or the code location package matches regexp
    82  func (lg *LogInstance) Debug(format string, a ...interface{}) {
    83  	var cloc *error116.CodeLocation
    84  	if !lg.isDebug.IsTrue() {
    85  		regExp := lg.getRegexp()
    86  		if regExp == nil {
    87  			return // debug: false regexp: nil
    88  		}
    89  		cloc = error116.NewCodeLocation(lg.stackFramesToSkip + logInstDebugFrameDelta)
    90  		if !regExp.MatchString(cloc.FuncName) {
    91  			return // debug: false regexp: no match
    92  		}
    93  	} else {
    94  		cloc = error116.NewCodeLocation(lg.stackFramesToSkip + logInstDebugFrameDelta)
    95  	}
    96  	if err := lg.output(0, appendLocation(sprintf(format, a...), cloc)); err != nil {
    97  		panic(Errorf("LogInstance output: %w", err))
    98  	}
    99  }

   118  func (lg *LogInstance) SetRegexp(regExp string) (err error) {
   119  	var regExpPt *regexp.Regexp
   120  	if regExp != "" {
   121  		if regExpPt, err = regexp.Compile(regExp); err != nil {
   122  			return Errorf("regexp.Compile: %w", err)
   123  		}
   124  	}
   125  	lg.infoLock.Lock()
   126  	defer lg.infoLock.Unlock()
   127  	lg.infoRegexp = regExpPt
   128  	return
   129  }

   140  // IsThisDebug returns whether debug logging is configured
   141  func (lg *LogInstance) IsThisDebug() bool {
   142  	if lg.isDebug.IsTrue() {
   143  		return true
   144  	}
   145  	regExp := lg.getRegexp()
   146  	if regExp == nil {
   147  		return false
   148  	}
   149  	cloc := error116.NewCodeLocation(lg.stackFramesToSkip - isThisDebugDelta)
   150  	return regExp.MatchString(cloc.FuncName)
   151  }

   158  func (lg *LogInstance) getRegexp() *regexp.Regexp {
   159  	lg.infoLock.RLock()
   160  	defer lg.infoLock.RUnlock()
   161  	return lg.infoRegexp

   164  func (lg *LogInstance) doLog(format string, a ...interface{}) {
   165  	s := sprintf(format, a...)
   166  	if lg.isDebug.IsTrue() {
   167  		s = appendLocation(s, error116.NewCodeLocation(lg.stackFramesToSkip))
   168  	}
   169  	if err := lg.output(0, s); err != nil {
   170  		panic(Errorf("LogInstance output: %w", err))
   171  	}
   172  }

   174  func appendLocation(s string, location *error116.CodeLocation) string {
   175  	// insert code location before a possible ending newline
   176  	sNewline := ""
   177  	if strings.HasSuffix(s, "\n") {
   178  		s = s[:len(s)-1]
   179  		sNewline = "\n"
   180  	}
   181  	return fmt.Sprintf("%s %s%s", s, location.Short(), sNewline)
   182  }
