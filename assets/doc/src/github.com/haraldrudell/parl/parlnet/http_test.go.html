    24  func TestHttpListen(t *testing.T) {
    25  	TCPaddress := "127.0.0.1:0"
    26  	network := ""
    27  	protocol := "http://"
    28  	URIPattern := "/" // "/" matches everything
    29  
    30  	t.Log("starting server")
    31  	hp := NewHttp(TCPaddress, network)
    32  	hp.HandleFunc(URIPattern, func(w http.ResponseWriter, req *http.Request) {
    33  		t.Logf("server received request from: %s", req.RemoteAddr)
    34  		w.WriteHeader(http.StatusNoContent)
    35  	})
    36  	errCh := hp.Listen()
    37  
    38  	// listen for errors
    39  	var errChWg sync.WaitGroup
    40  	errChWg.Add(1)
    41  	go func() {
    42  		defer errChWg.Done()
    43  		t.Log("Reading errCh")
    44  		for {
    45  			err, ok := <-errCh
    46  			if !ok {
    47  				break // errCh closed
    48  			}
    49  			t.Errorf("errCh: %+v", err)
    50  			panic(err)
    51  		}
    52  	}()
    53  
    54  	expectServerError := false
    55  	for once := true; once; once = false {
    56  
    57  		t.Logf("waiting for server ready")
    58  		isUp, addr := hp.WaitForUp()
    59  		if !isUp {
    60  			t.Error("Server failed to start")
    61  			expectServerError = true
    62  			break
    63  		}
    64  		t.Logf("server listening at %s", addr.String())
    65  
    66  		t.Logf("Sending http request")
    67  		requestURL := protocol + addr.String()
    68  		resp, err := http.Get(requestURL)
    69  		if err != nil {
    70  			t.Errorf("http.Get '%s': err: %+v", requestURL, err)
    71  			expectServerError = true
    72  			break
    73  		}
    74  		if resp.StatusCode != http.StatusNoContent {
    75  			t.Errorf("http.Get bad status code '%s': %d expected: %d", requestURL, resp.StatusCode, http.StatusNoContent)
    76  			return
    77  		}
    78  		if err := resp.Body.Close(); err != nil {
    79  			t.Errorf("Body.Close: %+v", err)
    80  			return
    81  		}
    82  		t.Logf("response status code: %d", resp.StatusCode)
    83  	}
    84  
    85  	// shutdown server
    86  	t.Logf("Shutting down server")
    87  	hp.Shutdown()
    88  
    89  	errChWg.Wait() // wait for errCh to close
    90  
    91  	if expectServerError {
    92  		panic(error116.New("Server provided no error")) // errCh should have paniced
    93  	}
    94  
    95  	t.Logf("Completed successfully")
    96  }
