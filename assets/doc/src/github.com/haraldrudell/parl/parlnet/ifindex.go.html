     6  package parlnet
     7  
     8  import (
     9  	"net"
    10  	"strconv"
    11  )
    12  
    13  // IfIndex is a dynamic reference to a network interface on Linux systems
    14  type IfIndex int
    15  
    16  // Present determines if intreface index is set
    17  func (ifIndex IfIndex) Present() bool {
    18  	return ifIndex > 0
    19  }
    20  
    21  // Interface gets net.Interface for ifIndex
    22  func (ifIndex IfIndex) Interface() (*net.Interface, error) {
    23  	return net.InterfaceByIndex(int(ifIndex))
    24  }
    25  
    26  // Zone gets net.IPAddr.Zone string for ifIndex
    27  func (ifIndex IfIndex) Zone() (zone string, err error) {
    28  	if ifIndex.Present() {
    29  		if iface, err := ifIndex.Interface(); err == nil { // may fail if interface already deleted
    30  			zone = iface.Name
    31  		} else {
    32  			zone = strconv.Itoa(int(ifIndex))
    33  		}
    34  	}
    35  	return
    36  }
