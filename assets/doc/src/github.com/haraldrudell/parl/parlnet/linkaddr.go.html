    14  // LinkAddr contains an Ethernet mac address, its interface name and interface index
    15  type LinkAddr struct {
    16  	Index IfIndex          // 0 is none
    17  	Name  string           // "" none
    18  	Hw    net.HardwareAddr // []byte
    19  }
    20  
    21  // NewLinkAddr instantiates LinkAddr
    22  func NewLinkAddr(index IfIndex) *LinkAddr {
    23  	return &LinkAddr{Index: index}
    24  }
    25  
    26  // NewLinkAddr2 instantiates LinkAddr
    27  func NewLinkAddr2(index IfIndex, name string, hw net.HardwareAddr) *LinkAddr {
    28  	return &LinkAddr{Index: index, Name: name, Hw: hw}
    29  }
    30  
    31  // UpdateName attempts to populate interface name if not already present
    32  func (a *LinkAddr) UpdateName() (err error) {
    33  	if a.Name == "" {
    34  		a.Name, err = a.Index.Zone()
    35  	}
    36  	return
    37  }
    38  
    39  // ZoneID is the IPv6 ZoneID for this interface
    40  func (a *LinkAddr) ZoneID() string {
    41  	if a != nil {
    42  		if a.Name != "" {
    43  			return a.Name
    44  		} else if a.Index > 0 {
    45  			return strconv.Itoa(int(a.Index))
    46  		}
    47  	}
    48  	return "0"
    49  }
    50  
    51  // OneString picks the most meaningful value
    52  func (a *LinkAddr) OneString() string {
    53  	if a != nil {
    54  		if a.Name != "" {
    55  			return a.Name
    56  		} else if len(a.Hw) > 0 {
    57  			return a.Hw.String()
    58  		} else if a.Index > 0 {
    59  			return strconv.Itoa(int(a.Index))
    60  		}
    61  	}
    62  	return "0"
    63  }
    64  
    65  func (a LinkAddr) String() string {
    66  	return fmt.Sprintf("%s(%d)%s", a.Name, a.Index, a.Hw)
    67  }
