    23  	tcp4 = "tcp4" // net network tcp ipv4

    26  // ListenTCP4 listens on local network interfaces with ipv4 tcp socket: ":8080" or "1.2.3.4:80"
    27  func ListenTCP4(socketString string) (socket *Socket, err error) {
    28  	var s Socket
    29  	var tcpAddr *net.TCPAddr
    30  	if tcpAddr, err = net.ResolveTCPAddr(tcp4, socketString); err != nil {
    31  		err = error116.Errorf("ResolveTCPAddr: '%w'", err)
    32  		return
    33  	}
    34  	if s.TCPListener, err = net.ListenTCP(tcp4, tcpAddr); err != nil {
    35  		err = error116.Errorf("ListenTCP: '%w'", err)
    36  		return
    37  	}
    38  	socket = &s
    39  	return
    40  }

    42  // RunHandler handles inbound connections
    43  func (s *Socket) RunHandler(handler func(net.Conn)) (errCh <-chan error) {
    44  	errChan := make(chan error)
    45  	s.wg.Add(1)
    46  	go s.accept(handler, errChan)
    47  	return errChan
    48  }

    50  func (s *Socket) accept(handler func(net.Conn), errCh chan<- error) {
    51  	defer close(errCh)
    52  	defer s.wg.Done()
    53  	parl.Recover2(parl.Annotation(), nil, func(e error) { errCh <- e })
    54  	for {
    55  		var err error
    56  		var conn net.Conn
    57  		if conn, err = s.Accept(); err != nil { // blocking: either a connection or an error
    58  			if opError, ok := err.(*net.OpError); ok {
    59  				if opError.Op == "accept" {
    60  					break // ListenTCP4 is closed
    61  				}
    62  			}
    63  			errCh <- parl.Errorf("TCPListener.Accept: %T '%[1]w'", err) // some error
    64  			continue
    65  		}
    66  
    67  		// invoke connection handler
    68  		s.wg.Add(1)
    69  		go func() {
    70  			defer s.wg.Done()
    71  			handler(conn)
    72  		}()
    73  	}
    74  }
    75  
    76  // Wait waits for all connections and the handler thread to exit. ListenTCP4.Close needs to be invoked
    77  func (s *Socket) Wait() {
    78  	s.wg.Wait()
    79  }
