    13  // NextHop describes a route target
    14  type NextHop struct {
    15  	/*
    16  		if NextHop is an address on the local host or on a local subnet,
    17  		Gateway is nil
    18  		LinkAddr describes the local interface
    19  		Src is the address on that local interface
    20  
    21  		If Nexthop is remote, beyond any local subnet,
    22  		Gateway is an IP on a local subnet
    23  		LinkAddr describes the local interface for that subnet
    24  		Src is the address on that local interface
    25  	*/
    26  	Gateway net.IPAddr
    27  	LinkAddr
    28  	Src net.IPAddr // the source ip to use on LinkAddr
    29  }

    31  // NewNextHop assembles a route destination
    32  func NewNextHop(gateway *net.IPAddr, linkAddr *LinkAddr, src *net.IPAddr) *NextHop {
    33  	next0 := NextHop{}
    34  	if gateway != nil {
    35  		next0.Gateway = *gateway
    36  	}
    37  	if linkAddr != nil {
    38  		next0.LinkAddr = *linkAddr
    39  	}
    40  	if src != nil {
    41  		next0.Src = *src
    42  	}
    43  	return &next0
    44  }

    46  // NewNextHop2 assembles a route destination based on IfIndex
    47  func NewNextHop2(index IfIndex, gateway *net.IPAddr, src *net.IPAddr) (next *NextHop, err error) {
    48  	var linkAddr *LinkAddr
    49  	if index.Present() {
    50  		linkAddr = NewLinkAddr(index)
    51  		if err = linkAddr.UpdateName(); err != nil {
    52  			return
    53  		}
    54  	}
    55  	return NewNextHop(gateway, linkAddr, src), err
    56  }

    73  // Target describes the detination for this next hop
    74  func (nh *NextHop) Target() (gateway *net.IPAddr, s string) {
    75  	if nh == nil {
    76  		return
    77  	}
    78  	s = nh.LinkAddr.OneString()
    79  	if !nh.HasGateway() {
    80  		return
    81  	}
    82  	gw := nh.Gateway
    83  	gateway = &gw
    84  	if gateway == nil {
    85  		return
    86  	}
    87  	srcIP := nh.Src.IP
    88  	s1 := srcIP.String()
    89  	index := nh.LinkAddr.Index
    90  	if index > 0 {
    91  		iface, err := net.InterfaceByIndex(int(index))
    92  		if err == nil {
    93  			addrs, e2 := iface.Addrs()
    94  			if e2 == nil {
    95  				if !IsNzIP(srcIP) {
    96  					if len(addrs) > 0 {
    97  						s1 = addrs[0].String()
    98  					}
    99  				} else {
   100  					for _, ipNet := range addrs {
   101  						if ipNet, ok := ipNet.(*net.IPNet); ok {
   102  							if ipNet.IP.Equal(srcIP) {
   103  								s1 = ipNet.String()
   104  								break
   105  							}
   106  						}
   107  					}
   108  				}
   109  			}
   110  		}
   111  	}
   112  	s = fmt.Sprintf("%s %s %s", gateway, s, s1)
   113  	return
   114  }
