    27  func NewHttps(host, network string, certDER parlca.CertificateDER, signer crypto.Signer) (hp *Https) {
    28  	if host == "" {
    29  		host = httpsAddr
    30  	}
    31  	return &Https{
    32  		Http:    *NewHttp(host, network),
    33  		Cert:    certDER,
    34  		Private: signer,
    35  	}
    36  }

    49  func (hp *Https) listenerThread() {
    50  	defer hp.CloseErr()
    51  	defer parl.Recover(parl.Annotation(), nil, hp.SendErr)
    52  	var didReadyWg bool
    53  	defer func() {
    54  		if !didReadyWg {
    55  			hp.ReadyWg.Done()
    56  		}
    57  	}()
    58  
    59  	tlsListener, err := hp.TLS() // *tls.listener — need this or it’s file certificates
    60  	if err != nil {
    61  		return
    62  	}
    63  	hp.ReadyWg.Done()
    64  	didReadyWg = true
    65  	hp.IsListening.Set()
    66  
    67  	srv := &hp.Server
    68  	if err := srv.Serve(tlsListener); err != nil { // blocking until Shutdown or Close
    69  		if err != http.ErrServerClosed {
    70  			hp.SendErr(error116.Errorf("srv.ServeTLS: '%w'", err))
    71  			return
    72  		}
    73  	}
    74  }

    76  func (hp *Https) TLS() (tlsListener net.Listener, err error) {
    77  	srv := &hp.Server
    78  
    79  	// make ServeTLS execute srv.setupHTTP2_ServeTLS go 1.16.6
    80  	badPath := "%"
    81  	if err = srv.ServeTLS(nil, badPath, badPath); err == nil {
    82  		err = error116.New("missing error from srv.ServeTLS")
    83  		return
    84  	} else {
    85  		if pathError, ok := err.(*fs.PathError); ok {
    86  			if pathError.Path == badPath {
    87  				err = nil
    88  			}
    89  		}
    90  		if err != nil {
    91  			err = error116.Errorf("srv.ServeTLS: '%w'", err)
    92  			return
    93  		}
    94  	}
    95  
    96  	// get *net.TCPListener from srv.ListenAndServeTLS
    97  	var listener net.Listener
    98  	if listener, err = hp.Listener(); err != nil {
    99  		return
   100  	}
   101  
   102  	// create a TLS listener from srv.ServeTLS
   103  	var tlsConfig *tls.Config
   104  	if srv.TLSConfig == nil {
   105  		tlsConfig = &tls.Config{}
   106  	} else {
   107  		tlsConfig = srv.TLSConfig.Clone()
   108  	}
   109  	if !parls.StrSliceContains(tlsConfig.NextProtos, http11) {
   110  		tlsConfig.NextProtos = append(tlsConfig.NextProtos, http11)
   111  	}
   112  	tlsConfig.Certificates = make([]tls.Certificate, 1)
   113  	tlsCertificate := &tlsConfig.Certificates[0]
   114  	tlsCertificate.Certificate = append(tlsCertificate.Certificate, hp.Cert) // certificate not from file system
   115  	tlsCertificate.PrivateKey = hp.Private                                   // private key not from file system
   116  	tlsListener = tls.NewListener(listener, tlsConfig)
   117  	return
   118  }
