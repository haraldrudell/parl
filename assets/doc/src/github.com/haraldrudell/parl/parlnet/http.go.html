    19  type Http struct {
    20  	Network string // "tcp", "tcp4", "tcp6", "unix" or "unixpacket"

    32  // NewHttp creates http server host is host:port, default ":http"
    33  func NewHttp(host, network string) (hp *Http) {
    34  	if host == "" {
    35  		host = httpAddr
    36  	}
    37  	if network == "" {
    38  		network = TCPNetwork
    39  	}
    40  	h := Http{
    41  		Network: network,
    42  		Server: http.Server{
    43  			Handler: http.NewServeMux(), // struct
    44  			Addr:    host,
    45  		},
    46  	}
    47  	return &h
    48  }

    58  func (hp *Http) HandleFunc(pattern string, handler HandlerFunc) {
    59  	var httpServeMux *http.ServeMux
    60  	var ok bool
    61  	if httpServeMux, ok = hp.Handler.(*http.ServeMux); !ok {
    62  		panic(error116.Errorf("server.Handler not *http.ServeMux: %T", hp.Handler))
    63  	}
    64  	httpServeMux.HandleFunc(pattern, handler)
    65  }

    67  func (hp *Http) Listen() (errCh <-chan error) {
    68  	errCh = hp.SubListen()
    69  	go hp.listenerThread()
    70  	return
    71  }

    73  func (hp *Http) SubListen() (errCh <-chan error) {
    74  	hp.ReadyWg.Add(1)
    75  	defer hp.ReadyWg.Done()
    76  	if !hp.ListenInvoked.Set() {
    77  		panic(error116.New("multiple http.Run invocations"))
    78  	}
    79  	errChan := make(chan error)
    80  	errCh = errChan
    81  	hp.ErrCh = errChan
    82  	hp.ReadyWg.Add(1)
    83  	return
    84  }
    85  
    86  func (hp *Http) listenerThread() {
    87  	defer hp.CloseErr()
    88  	defer parl.Recover(parl.Annotation(), nil, hp.SendErr)
    89  	var didReadyWg bool
    90  	defer func() {
    91  		if !didReadyWg {
    92  			hp.ReadyWg.Done()
    93  		}
    94  	}()
    95  
    96  	listener, err := hp.Listener()
    97  	if err != nil {
    98  		return
    99  	}
   100  	hp.ReadyWg.Done()
   101  	didReadyWg = true
   102  	hp.IsListening.Set()
   103  
   104  	if err := hp.Server.Serve(listener); err != nil { // blocking until Shutdown or Close
   105  		if err != http.ErrServerClosed {
   106  			hp.SendErr(error116.Errorf("hp.Server.Serve: '%w'", err))
   107  			return
   108  		}
   109  	}
   110  }

   112  func (hp *Http) Listener() (listener net.Listener, err error) {
   113  	srv := &hp.Server
   114  	listener, err = net.Listen(hp.Network, srv.Addr)
   115  	if err != nil {
   116  		hp.SendErr(error116.Errorf("net.Listen %s %s: '%w'", hp.Network, srv.Addr, err))
   117  		return
   118  	}
   119  	hp.Addr = listener.Addr()
   120  	return
   121  }

   123  func (hp *Http) WaitForUp() (isUp bool, addr net.Addr) {
   124  	if !hp.ListenInvoked.IsTrue() {
   125  		return // Listen has not been invoked
   126  	}
   127  	hp.ReadyWg.Wait()
   128  	if isUp = hp.IsListening.IsTrue(); isUp {
   129  		addr = hp.Addr
   130  	}
   131  	return
   132  }

   134  func (hp *Http) SendErr(err error) {
   135  	hp.ErrChMutex.Lock()
   136  	defer hp.ErrChMutex.Unlock()
   137  	if !hp.ErrChClosed.IsTrue() {
   138  		hp.ErrCh <- err
   139  	}
   140  }
   141  
   142  func (hp *Http) CloseErr() {
   143  	if hp.ErrChClosed.Set() {
   144  		hp.ErrChMutex.Lock()
   145  		close(hp.ErrCh)
   146  		hp.ErrChMutex.Unlock()
   147  	}
   148  }
   149  
   150  func (hp *Http) Shutdown() {
   151  	if !hp.IsShutdown.Set() {
   152  		return // already shutdown
   153  	}
   154  	ctx, cancel := context.WithTimeout(context.Background(), httpShutdownTimeout)
   155  	defer cancel()
   156  	if err := hp.Server.Shutdown(ctx); err != nil {
   157  		hp.SendErr(error116.Errorf("hp.Server.Shutdown: '%w'", err))
   158  	}
   159  }
