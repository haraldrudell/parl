    21  func TestSocket(t *testing.T) {
    22  	socketString := "127.0.0.1:0" // 0 means ephemeral port
    23  	colonPos := strings.Index(socketString, ":")
    24  	if colonPos == -1 {
    25  		t.Errorf("Bad socketString fixture: %q", socketString)
    26  		return
    27  	}
    28  	socket, err := ListenTCP4(socketString)
    29  	if err != nil {
    30  		t.Errorf("ListenTCP4 error: %+v", err)
    31  		return
    32  	}
    33  	addr := socket.Addr().String()
    34  	if !strings.HasPrefix(addr, socketString[:colonPos+1]) {
    35  		t.Errorf("Bad socket adress: %q", addr)
    36  		return
    37  	}
    38  	if port, err := strconv.Atoi(strings.TrimPrefix(addr, socketString[:colonPos+1])); err != nil {
    39  		t.Errorf("Bad port number: %q", addr)
    40  		return
    41  	} else if port < 1 || port > 65535 {
    42  		t.Errorf("Bad port numeric: %v", port)
    43  		return
    44  	}
    45  	if err := socket.Close(); err != nil {
    46  		t.Errorf("socket.Close: '%w'", err)
    47  		return
    48  	}
    49  }

    51  func TestRunHandler(t *testing.T) {
    52  
    53  	// set up socket
    54  	socketString := "127.0.0.1:0" // 0 means ephemeral port
    55  	socket, err := ListenTCP4(socketString)
    56  	if err != nil {
    57  		t.Errorf("ListenTCP4 error: %+v", err)
    58  		return
    59  	}
    60  
    61  	// connection listener
    62  	var count int64
    63  	connFunc := func(conn net.Conn) {
    64  		if err := conn.Close(); err != nil {
    65  			panic(parl.Errorf("conn.Close: '%w'", err))
    66  		}
    67  		atomic.AddInt64(&count, 1)
    68  	}
    69  	errCh := socket.RunHandler(connFunc)
    70  	go func() {
    71  		for {
    72  			if err, ok := <-errCh; !ok {
    73  				break
    74  			} else {
    75  				t.Logf("errCh: %+v", err)
    76  			}
    77  		}
    78  	}()
    79  
    80  	// connect to socket
    81  	ctx := context.Background()
    82  	addr := socket.Addr()
    83  	var tcpClient net.Dialer
    84  	conn, err := tcpClient.DialContext(ctx, addr.Network(), addr.String())
    85  	if err != nil {
    86  		t.Errorf("tcpClient.DialContext: '%w'", err)
    87  		return
    88  	}
    89  
    90  	// read from socket
    91  	bytes := make([]byte, 1)
    92  	for {
    93  		n, err := conn.Read(bytes)
    94  		if err != nil {
    95  			if errors.Is(err, io.EOF) {
    96  				break
    97  			} else {
    98  				t.Errorf("conn.Read: '%w'", err)
    99  				return
   100  			}
   101  		}
   102  		if n != 0 {
   103  			t.Errorf("conn.Read unexpected bytes: %d", n)
   104  			return
   105  		}
   106  	}
   107  
   108  	// close client
   109  	if err := conn.Close(); err != nil {
   110  		t.Errorf("client Close: '%w'", err)
   111  		return
   112  	}
   113  
   114  	// close listener
   115  	if err := socket.Close(); err != nil {
   116  		t.Errorf("client Close: '%w'", err)
   117  		return
   118  	}
   119  
   120  	t.Logf("socket.Wait… %d", atomic.LoadInt64(&count))
   121  	socket.Wait()
   122  }
