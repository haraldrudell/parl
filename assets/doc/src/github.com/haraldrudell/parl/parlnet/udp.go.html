    33  // NewUDP network: "udp" "udp4" "udp6" address: "host:port"
    34  func NewUDP(network, address string, udpFunc UDPFunc, maxSize int) (udp *UDP) {
    35  	if maxSize < 1 {
    36  		maxSize = udpDefaultMaxSize
    37  	}
    38  	if netUDPAddr, err := net.ResolveUDPAddr(network, address); err != nil {
    39  		panic(error116.Errorf("net.ResolveUDPAddr: '%w'", err))
    40  	} else {
    41  		udp = &UDP{UDPAddr: *netUDPAddr, Network: network, F: udpFunc, MaxSize: maxSize}
    42  	}
    43  	return
    44  }

    53  func (udp *UDP) Listen() (errCh <-chan error) {
    54  	udp.StartingListen.Add(1)
    55  	if !udp.ListenInvoked.Set() {
    56  		udp.StartingListen.Done()
    57  		panic(error116.New("multiple udp.Listen invocations"))
    58  	}
    59  	if udp.IsShutdown.IsTrue() {
    60  		udp.StartingListen.Done()
    61  		panic(error116.New("udp.Listen after Shutdown"))
    62  	}
    63  	errChan := make(chan error)
    64  	errCh = errChan
    65  	udp.ErrCh = errChan
    66  	go udp.listenThread()
    67  	return
    68  }

    70  func (udp *UDP) listenThread() {
    71  	errCh := udp.ErrCh
    72  	defer close(errCh)
    73  	var FInvocations sync.WaitGroup
    74  	defer FInvocations.Wait()
    75  	var startingDone bool
    76  	defer func() {
    77  		if !startingDone {
    78  			udp.StartingListen.Done()
    79  		}
    80  	}()
    81  	defer parl.Recover2(parl.Annotation(), nil, func(e error) { errCh <- e }) // capture panics
    82  
    83  	// listen
    84  	var netUDPConn *net.UDPConn // represents a network file descriptor
    85  	var err error
    86  	if netUDPConn, err = net.ListenUDP(udp.Network, &udp.UDPAddr); err != nil {
    87  		errCh <- error116.Errorf("net.ListenUDP: '%w'", err)
    88  		return
    89  	}
    90  	if udp.setConn(netUDPConn) { // isShutdown
    91  		if err = netUDPConn.Close(); err != nil {
    92  			errCh <- error116.Errorf("netUDPConn.Close: '%w'", err)
    93  		}
    94  		return
    95  	}
    96  	udp.Addr = netUDPConn.LocalAddr()
    97  	udp.IsListening.Set()
    98  	udp.StartingListen.Done()
    99  	startingDone = true
   100  	defer func() {
   101  		if !udp.IsShutdown.IsTrue() {
   102  			if err := netUDPConn.Close(); err != nil {
   103  				errCh <- err
   104  			}
   105  		}
   106  		udp.IsListening.Clear()
   107  	}()
   108  
   109  	// read datagrams
   110  	for {
   111  		b := make([]byte, udp.MaxSize)
   112  		oob := make([]byte, oobSize)
   113  		var n int
   114  		var oobn int
   115  		var flags int
   116  		var addr *net.UDPAddr
   117  		var err error
   118  		n, oobn, flags, addr, err = netUDPConn.ReadMsgUDP(b, oob)
   119  		if err != nil {
   120  			if udp.IsShutdown.IsTrue() && udp.isClosedErr(err) {
   121  				return // we are shutdown
   122  			}
   123  			errCh <- error116.Errorf("ReadMsgUDP: '%w'", err)
   124  			return
   125  		}
   126  		FInvocations.Add(1)
   127  		go func() {
   128  			defer FInvocations.Done()
   129  			udp.F(b[:n], oob[:oobn], flags, addr)
   130  		}()
   131  	}
   132  }

   134  func (udp *UDP) WaitForUp() (isUp bool, addr net.Addr) {
   135  	if !udp.ListenInvoked.IsTrue() {
   136  		return // Listen has not been invoked
   137  	}
   138  	udp.StartingListen.Wait()
   139  	if isUp = udp.IsListening.IsTrue(); isUp {
   140  		addr = udp.Addr
   141  	}
   142  	return
   143  }

   145  func (udp *UDP) isClosedErr(err error) (isClose bool) {
   146  	// read udp 127.0.0.1:50050: use of closed network connection
   147  	// &net.OpError{Op:"read", Net:"udp", Source:(*net.UDPAddr)(0xc00007a030), Addr:net.Addr(nil), Err:(*errors.errorString)(0xc000098160)}
   148  	opErr, ok := err.(*net.OpError)
   149  	if !ok {
   150  		return
   151  	}
   152  	if opErr.Op != netReadOperation {
   153  		return
   154  	}
   155  	e := opErr.Err
   156  	// &errors.errorString{s:"use of closed network connection"}
   157  	if e.Error() != useOfClosed {
   158  		return // some other error
   159  	}
   160  	isClose = true
   161  	return
   162  }

   164  func (udp *UDP) setConn(conn *net.UDPConn) (isShutdown bool) {
   165  	udp.connMutex.Lock()
   166  	defer udp.connMutex.Unlock()
   167  	isShutdown = udp.IsShutdown.IsTrue()
   168  	if !isShutdown {
   169  		udp.NetUDPConn = conn
   170  	}
   171  	return
   172  }
   173  
   174  func (udp *UDP) Shutdown() {
   175  	udp.connMutex.RLock()
   176  	defer udp.connMutex.RUnlock()
   177  	if !udp.IsShutdown.Set() {
   178  		return // it was already shutdown
   179  	}
   180  	conn := udp.NetUDPConn
   181  	if conn == nil {
   182  		return // no need to close connection
   183  	}
   184  	err := conn.Close()
   185  	if err == nil {
   186  		return // conn successfully closed
   187  	}
   188  	udp.ErrCh <- err
   189  }
