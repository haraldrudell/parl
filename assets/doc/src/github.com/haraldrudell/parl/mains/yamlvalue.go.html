    15  type YamlValue struct {
    16  	Name    string
    17  	Pointer interface{}
    18  }

    20  func NewYamlValue(instancePointer interface{}, fieldPointer interface{}) (yv *YamlValue) {
    21  	reflectValue := reflect.ValueOf(instancePointer)
    22  	if !reflectValue.IsValid() {
    23  		parl.Errorf("NewYamlValue: instancePointer cannot be nil")
    24  	}
    25  	structValue := reflectValue.Elem() // retrieve what the pointer points to
    26  	if structValue.Kind() != reflect.Struct {
    27  		parl.Errorf("NewYamlValue: instancePointer not pointer to struct instance")
    28  	}
    29  	numField := structValue.NumField()
    30  	for i := 0; i < numField; i++ {
    31  		fieldValue := structValue.Field(i)
    32  		if !fieldValue.CanAddr() {
    33  			parl.Errorf("NewYamlValue: field#%d: canAddr false", i)
    34  		}
    35  		fieldAddr := fieldValue.Addr()
    36  		if !fieldAddr.CanInterface() {
    37  			parl.Errorf("NewYamlValue: field#%d: canInterface false", i)
    38  		}
    39  		ifValue := fieldAddr.Interface()
    40  		if ifValue == fieldPointer {
    41  			return &YamlValue{
    42  				Name:    strings.ToLower(structValue.Type().Field(i).Name),
    43  				Pointer: fieldPointer}
    44  		}
    45  	}
    46  	parl.Errorf("NewYamlValue: fieldPointer not found")
    47  	return
    48  }
