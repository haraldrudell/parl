     8  import (
     9  	"fmt"
    10  	"strings"
    11  )
    12  
    13  const (
    14  	statusLeadin = "\x20\x20"
    15  	leadinLen    = len(statusLeadin)
    16  )
    17  
    18  // GetStatus gets instance of status printer
    19  func GetStatus() (stat *Status) {
    20  	stat = &Status{}
    21  	return
    22  }
    23  
    24  // Status handles printing of status to stdout
    25  type Status struct {
    26  	chars uint
    27  }
    28  
    29  // Print status information
    30  func (stat *Status) Print(s string) {
    31  	chars := len(s)
    32  	spaceChars := int(stat.chars) - chars
    33  	if spaceChars < 0 {
    34  		spaceChars = 0
    35  	}
    36  	if chars == 0 && spaceChars == 0 {
    37  		return
    38  	}
    39  	stat.chars = uint(chars)
    40  	fmt.Printf("%s%s%s%s", statusLeadin, s,
    41  		strings.Repeat("\x20", spaceChars),
    42  		strings.Repeat("\b", chars+spaceChars+leadinLen))
    43  }
    44  
    45  // Clear remove previously printed status
    46  func (stat *Status) Clear() {
    47  	stat.Print("")
    48  }
