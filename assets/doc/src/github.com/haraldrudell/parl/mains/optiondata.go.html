    30  // AddOption executes flag.BoolVar and such on the options map
    31  func (om *OptionData) AddOption() {
    32  	switch p := om.P.(type) {
    33  	case *bool:
    34  		if value, ok := om.Value.(bool); !ok {
    35  			panic(fmt.Errorf("option '%s' Boolean default value not boolean: %T", om.Name, om.Value))
    36  		} else {
    37  			flag.BoolVar(p, om.Name, value, om.Usage)
    38  		}
    39  	case *time.Duration:
    40  		if value, ok := om.Value.(time.Duration); !ok {
    41  			panic(fmt.Errorf("option '%s' Duration default value not duration: %T", om.Name, om.Value))
    42  		} else {
    43  			flag.DurationVar(p, om.Name, value, om.Usage)
    44  		}
    45  	case *float64:
    46  		if value, ok := om.Value.(float64); !ok {
    47  			panic(fmt.Errorf("option '%s' float64 default value not float64: %T", om.Name, om.Value))
    48  		} else {
    49  			flag.Float64Var(p, om.Name, value, om.Usage)
    50  		}
    51  	case *int64:
    52  		if value, ok := om.Value.(int64); !ok {
    53  			panic(fmt.Errorf("option '%s' Int64 default value not int64: %T", om.Name, om.Value))
    54  		} else {
    55  			flag.Int64Var(p, om.Name, value, om.Usage)
    56  		}
    57  	case *int:
    58  		if value, ok := om.Value.(int); !ok {
    59  			panic(fmt.Errorf("option '%s' Int default value not int: %T", om.Name, om.Value))
    60  		} else {
    61  			flag.IntVar(p, om.Name, value, om.Usage)
    62  		}
    63  	case *string:
    64  		if value, ok := om.Value.(string); !ok {
    65  			panic(fmt.Errorf("option '%s' String default value not string: %T", om.Name, om.Value))
    66  		} else {
    67  			flag.StringVar(p, om.Name, value, om.Usage)
    68  		}
    69  	case *uint64:
    70  		if value, ok := om.Value.(uint64); !ok {
    71  			panic(fmt.Errorf("option '%s' Uint64 default value not uint64: %T", om.Name, om.Value))
    72  		} else {
    73  			flag.Uint64Var(p, om.Name, value, om.Usage)
    74  		}
    75  	case *uint:
    76  		if value, ok := om.Value.(uint); !ok {
    77  			panic(fmt.Errorf("option '%s' Uint default value not uint: %T", om.Name, om.Value))
    78  		} else {
    79  			flag.UintVar(p, om.Name, value, om.Usage)
    80  		}
    81  	case *[]string:
    82  		if value, ok := om.Value.([]string); !ok {
    83  			panic(fmt.Errorf("option '%s' []string default value not []string: %T", om.Name, om.Value))
    84  		} else {
    85  			flag.Var(GetStringSliceValue(p, value), om.Name, om.Usage)
    86  		}
    87  	default:
    88  		panic(fmt.Errorf("option '%s' Unknown options type: %T", om.Name, p))
    89  	}
    90  }

    96  func (om *OptionData) ApplyYaml() (err error) {
    97  	if om.Y == nil {
    98  		return // does not have YamlValue
    99  	}
   100  	yamlDataPointer := om.Y.Pointer
   101  	parl.Debug("optionData.ApplyYaml option: -%s yaml key:%q", om.Name, om.Y.Name)
   102  	switch valuePointer := om.P.(type) {
   103  	case *bool:
   104  		typedPointer, ok := yamlDataPointer.(*bool)
   105  		if !ok {
   106  			return parl.Errorf(typeMismatch, om.Name, om.P, yamlDataPointer)
   107  		}
   108  		*valuePointer = *typedPointer
   109  	case *time.Duration:
   110  		typedPointer, ok := yamlDataPointer.(*time.Duration)
   111  		if !ok {
   112  			return parl.Errorf(typeMismatch, om.Name, om.P, yamlDataPointer)
   113  		}
   114  		*valuePointer = *typedPointer
   115  	case *float64:
   116  		typedPointer, ok := yamlDataPointer.(*float64)
   117  		if !ok {
   118  			return parl.Errorf(typeMismatch, om.Name, om.P, yamlDataPointer)
   119  		}
   120  		*valuePointer = *typedPointer
   121  	case *int64:
   122  		typedPointer, ok := yamlDataPointer.(*int64)
   123  		if !ok {
   124  			return parl.Errorf(typeMismatch, om.Name, om.P, yamlDataPointer)
   125  		}
   126  		*valuePointer = *typedPointer
   127  	case *int:
   128  		typedPointer, ok := yamlDataPointer.(*int)
   129  		if !ok {
   130  			return parl.Errorf(typeMismatch, om.Name, om.P, yamlDataPointer)
   131  		}
   132  		*valuePointer = *typedPointer
   133  	case *string:
   134  		typedPointer, ok := yamlDataPointer.(*string)
   135  		if !ok {
   136  			return parl.Errorf(typeMismatch, om.Name, om.P, yamlDataPointer)
   137  		}
   138  		*valuePointer = *typedPointer
   139  	case *uint64:
   140  		typedPointer, ok := yamlDataPointer.(*uint64)
   141  		if !ok {
   142  			return parl.Errorf(typeMismatch, om.Name, om.P, yamlDataPointer)
   143  		}
   144  		*valuePointer = *typedPointer
   145  	case *uint:
   146  		typedPointer, ok := yamlDataPointer.(*uint)
   147  		if !ok {
   148  			return parl.Errorf(typeMismatch, om.Name, om.P, yamlDataPointer)
   149  		}
   150  		*valuePointer = *typedPointer
   151  	case *[]string:
   152  		typedPointer, ok := yamlDataPointer.(*[]string)
   153  		if !ok {
   154  			return parl.Errorf(typeMismatch, om.Name, om.P, yamlDataPointer)
   155  		}
   156  		*valuePointer = *typedPointer
   157  	default:
   158  		return parl.Errorf("option %s: unknown value type: %T", om.Name, om.P)
   159  	}
   160  	return
   161  }
