    68  // Init populate launch time — may throw
    69  func (ex *Executable) Init() *Executable {
    70  	now := time.Now()
    71  	ex.Launch = now
    72  	ex.LaunchString = now.Format(rfcTimeFormat)
    73  	ex.Host = parlos.ShortHostname()
    74  	if len(os.Args) > 1 && os.Args[1] == SilentString {
    75  		parl.SetSilent(true)
    76  	}
    77  	return ex
    78  }

   123  // PrintBannerAndParseOptions prints greeting
   124  func (ex *Executable) PrintBannerAndParseOptions(om []OptionData) (ex1 *Executable) {
   125  	ex1 = ex
   126  	// print program name and populated details
   127  	banner := parls.FilteredJoin([]string{
   128  		parls.FilteredJoin([]string{ex.Program, ex.Version, ex.Comment}, "\x20"),
   129  		ex.Copyright,
   130  		ex.License,
   131  		fmt.Sprintf(timeHeader, ex.LaunchString),
   132  		fmt.Sprintf(hostHeader, ex.Host),
   133  	}, "\n")
   134  	if len(banner) != 0 {
   135  		parl.Info(banner)
   136  	}
   137  
   138  	// parse options
   139  	flag.Usage = ex.usage
   140  	omLen := len(om)
   141  	for i := 0; i < omLen; i++ {
   142  		(&om[i]).AddOption()
   143  	}
   144  	flag.Parse()
   145  
   146  	// parse arguments
   147  	args := flag.Args() // command-line arguments not part of flags
   148  	count := len(args)
   149  	argsOk :=
   150  		count == 0 && (ex.Arguments&NoArguments != 0) ||
   151  			count == 1 && (ex.Arguments&OneArgument != 0) ||
   152  			count > 0 && (ex.Arguments&ManyArguments != 0)
   153  	if !argsOk {
   154  		if count == 0 {
   155  			if ex.Arguments&ManyArguments != 0 {
   156  				parl.Log("There must be one or more arguments")
   157  			} else {
   158  				parl.Log("There must be one argument")
   159  			}
   160  		} else {
   161  			for i, v := range args {
   162  				args[i] = fmt.Sprintf("%q", v)
   163  			}
   164  			parl.Log("Unknown parameters: %s\n", strings.Join(args, "\x20"))
   165  		}
   166  		ex.usage()
   167  		parlos.Exit(parlos.StatusCodeUsage, nil)
   168  	}
   169  	ex.ArgCount = count
   170  	if count == 1 && (ex.Arguments&OneArgument != 0) {
   171  		ex.Arg = args[0]
   172  	}
   173  	if count > 0 && (ex.Arguments&ManyArguments != 0) {
   174  		ex.Args = args
   175  	}
   176  	return
   177  }

   196  func (ex *Executable) ApplyYaml(yamlFile, yamlKey string, thunk UnmarshalThunk, om []OptionData) {
   197  	parl.Debug("exe.ApplyYaml: file: %q", yamlFile)
   198  	filename, bytes := FindFile(yamlFile, ex.Program)
   199  	if filename == "" || len(bytes) == 0 {
   200  		parl.Debug("ApplyYaml: no yaml file")
   201  		return
   202  	}
   203  	yamlDictionaryKey := GetTopLevelKey(yamlKey) // key name from option or a default
   204  
   205  	// try to obtain the list of defined keys in the options dictionary
   206  	var yamlVisitedKeys map[string]bool
   207  	yco := map[string]map[string]interface{}{}
   208  	parl.Debug("ApplyYaml: first yaml.Unmarshal")
   209  	if yaml.Unmarshal(bytes, &yco) == nil {
   210  		yamlVisitedKeys = map[string]bool{}
   211  		if optionsMap := yco[yamlDictionaryKey]; optionsMap != nil {
   212  			for key := range optionsMap {
   213  				yamlVisitedKeys[strings.ToLower(key)] = true
   214  			}
   215  		}
   216  	}
   217  	parl.Debug("ApplyYaml: yamlVisitedKeys: %v\n", yamlVisitedKeys)
   218  
   219  	hasData, err := thunk(bytes, yaml.Unmarshal, yamlDictionaryKey)
   220  	if err != nil {
   221  		ex.AddErr(parl.Errorf("ApplyYaml thunk: filename: %q: %w", filename, err)).Exit()
   222  	} else if !hasData {
   223  		return
   224  	}
   225  
   226  	// iterate over options
   227  	// ignore if no yaml key
   228  	// ignore if yamlVisitedKeys exists and do not have the option
   229  	visitedOptions := map[string]bool{}
   230  	flag.Visit(func(fp *flag.Flag) { visitedOptions[fp.Name] = true })
   231  	for _, optionData := range om {
   232  		if visitedOptions[optionData.Name] || // was specified on command line
   233  			optionData.Y == nil { // does not have yaml value
   234  			continue
   235  		}
   236  		if yamlVisitedKeys != nil { // we hve a map of the yaml keys present
   237  			if !yamlVisitedKeys[optionData.Y.Name] {
   238  				continue // this key was not present in yaml
   239  			}
   240  		} else if parls.IsDefaultValue(optionData.Y.Pointer) {
   241  			continue // no visited information,, so ignore default values
   242  		}
   243  		if err := optionData.ApplyYaml(); err != nil {
   244  			ex.AddErr(err)
   245  			ex.Exit()
   246  		}
   247  	}
   248  }

   270  // Recover function to be deferred in main()
   271  func (ex *Executable) Recover() {
   272  	if e := recover(); e != nil {
   273  		hasStack := false
   274  		if err, ok := e.(error); ok {
   275  			hasStack = error116.HasStack(err)
   276  		}
   277  		parl.Debug("exe.Recover: executable %s panic: %T hasStack: %t '%+[2]v'", ex.Program, e, hasStack)
   278  		fmt.Println("Unhandled panic invoked exe.Recover: stack:")
   279  		debug.PrintStack()
   280  		var err error
   281  		var ok bool
   282  		if err, ok = e.(error); !ok {
   283  			err = fmt.Errorf("non-error value: %T %[1]v", e)
   284  		}
   285  		err = parl.Errorf("Unhandled panic to exe.Recover: '%w'", err)
   286  		ex.AddErr(err)
   287  		ex.Exit()
   288  	}
   289  }

   291  // AddErr extended with immediate printing of first error
   292  func (ex *Executable) AddErr(err error) (x *Executable) {
   293  	if parl.IsThisDebug() {
   294  		packFunc := error116.PackFunc()
   295  		var errS string
   296  		if err != nil {
   297  			errS = "\x27" + err.Error() + "\x27"
   298  		} else {
   299  			errS = "nil"
   300  		}
   301  		parl.Debug("%s(error: %s)\n%[1]s invocation: %[3]s", packFunc, errS, runt.Invocation(0))
   302  	}
   303  	x = ex
   304  	if err == nil {
   305  		return
   306  	}
   307  	if ex.err == nil {
   308  		ex.PrintErr(err)
   309  		ex.err = err
   310  		return
   311  	}
   312  	if errorList, ok := ex.err.(error116.ErrorHasList); ok {
   313  		errorList.Append(err)
   314  		return
   315  	}
   316  	ex.err = error116.AppendError(ex.err, err)
   317  	return
   318  }

   320  // PrintErr prints an error
   321  func (ex *Executable) PrintErr(err error) {
   322  	format := error116.DefaultFormat // format := "%v\n"
   323  	if ex.IsLongErrors {
   324  		format = error116.LongFormat // format = "%+v\n"
   325  	} else if ex.IsErrorLocation {
   326  		format = error116.ShortFormat
   327  	}
   328  	fmt.Fprintln(os.Stderr, error116.ChainString(err, format))
   329  }

   331  // Exit terminate from mains.err: exit 0 or echo to stderr and status code 1
   332  func (ex *Executable) Exit() {
   333  	if ex.err == nil {
   334  		parl.Debug("exe.Exit: no error")
   335  	} else {
   336  		parl.Debug("exe.Exit: err: %T '%[1]v'", ex.err)
   337  	}
   338  	parl.Debug("%s", debug.Stack())
   339  	if ex.err == nil {
   340  		parlos.Exit0()
   341  	}
   342  	errorList, isList := ex.err.(error116.ErrorHasList)
   343  	if isList {
   344  		for _, e := range errorList.ErrorList() {
   345  			ex.PrintErr(e)
   346  		}
   347  	}
   348  	if ex.IsLongErrors || isList {
   349  		fmt.Fprintln(os.Stderr, ex.err)
   350  	}
   351  	parlos.Exit1(nil)
   352  }
