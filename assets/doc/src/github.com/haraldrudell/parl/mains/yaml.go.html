    43  // FindFile locates and read the yaml file
    44  func FindFile(filename0, program string) (out string, bytes []byte) {
    45  	if filename0 != "" {
    46  		out = Abs(filename0)
    47  		bytes, _ = readFile(out)
    48  		return
    49  	}
    50  	dirs := []string{path.Join(parlos.UserHomeDir(), appsName), ParentDir(), Abs(etcName)}
    51  	filenames := []string{fmt.Sprintf("%s-%s%s", program, parlos.ShortHostname(), yamlExt),
    52  		program + yamlExt}
    53  	for _, dir := range dirs {
    54  		for _, f := range filenames {
    55  			filename := path.Join(dir, f)
    56  			var exists bool
    57  			if bytes, exists = readFile(filename); exists {
    58  				out = filename
    59  				return
    60  			}
    61  		}
    62  	}
    63  	return
    64  }
    65  
    66  func readFile(filename string) (bytes []byte, exists bool) {
    67  	var err error
    68  	if bytes, err = ioutil.ReadFile(filename); err != nil {
    69  		if !os.IsNotExist(err) {
    70  			panic(parl.Errorf("ioutil.ReadFile: '%w'", err))
    71  		}
    72  	} else {
    73  		exists = true
    74  	}
    75  	return
    76  }
