    15  func TestModerator(t *testing.T) {
    16  	doA := "a"
    17  	exitA := "A"
    18  	goB := "d"
    19  	doB := "b"
    20  	doC := "c"
    21  	goC := "¢"
    22  	exitB := "B"
    23  	exitC := "C"
    24  	expectedSequence := "a¢dAcCbB"
    25  	exitAIndex := strings.Index(expectedSequence, exitA)
    26  
    27  	Append := func() func(string) string {
    28  		sequence := ""
    29  		seqLock := sync.Mutex{}
    30  		return func(sIn string) (sOut string) {
    31  			seqLock.Lock()
    32  			defer seqLock.Unlock()
    33  			sequence += sIn
    34  			return sequence
    35  		}
    36  	}()
    37  
    38  	// new state
    39  	expectedStatusNew := "available: 1(1)"
    40  	mo := NewModerator(1)
    41  	actual := mo.String()
    42  	if actual != expectedStatusNew {
    43  		t.Logf("New status: expected: %s actual: %s", expectedStatusNew, actual)
    44  		t.Fail()
    45  	}
    46  
    47  	// state one: one thread running, no tickets left
    48  	expectedStatusOne := "waiting: 0(1)"
    49  	wgADo := sync.WaitGroup{}
    50  	wgADo.Add(1)
    51  	wgAWait := sync.WaitGroup{}
    52  	wgAWait.Add(1)
    53  	wgAExit := sync.WaitGroup{}
    54  	wgAExit.Add(1)
    55  	go func() {
    56  		if err := mo.Do(func() (err error) {
    57  			Append(doA)
    58  			wgADo.Done()
    59  			wgAWait.Wait()
    60  			return
    61  		}); err != nil {
    62  			t.Logf("A err: %v", err)
    63  			t.Fail()
    64  		}
    65  		Append(exitA)
    66  		wgAExit.Done()
    67  	}()
    68  	wgADo.Wait()
    69  	actual = mo.String()
    70  	if actual != expectedStatusOne {
    71  		t.Logf("State One: expected: %s actual: %s", expectedStatusOne, actual)
    72  		t.Fail()
    73  	}
    74  
    75  	// state two: 2 threads in queue
    76  	expectedStatusTwo := "waiting: 2(1)"
    77  	wgBCWait := sync.WaitGroup{}
    78  	wgBCWait.Add(1)
    79  	wgBCExit := sync.WaitGroup{}
    80  	wgBCExit.Add(2)
    81  	BC := func(goo, do, exit string) {
    82  		Append(goo)
    83  		if err := mo.Do(func() (err error) {
    84  			Append(do)
    85  			wgBCWait.Wait()
    86  			return
    87  		}); err != nil {
    88  			t.Logf("%s err: %v", do, err)
    89  		}
    90  		Append(exit)
    91  		wgBCExit.Done()
    92  	}
    93  	go BC(goB, doB, exitB)
    94  	go BC(goC, doC, exitC)
    95  	for {
    96  		actual = mo.String() // will hang if something wrong
    97  		if expectedStatusTwo == actual {
    98  			break
    99  		}
   100  		time.Sleep(10 * time.Millisecond)
   101  	}
   102  
   103  	// state three: 1 thread in queue
   104  	expectedStatusThree := "waiting: 1(1)"
   105  	wgAWait.Done()
   106  	wgAExit.Wait()
   107  	actual = mo.String()
   108  	if actual != expectedStatusThree {
   109  		t.Logf("Status Three: expected: %s actual: %s", expectedStatusThree, actual)
   110  		t.Fail()
   111  	}
   112  
   113  	// state 4: 1 ticket available
   114  	expectedStatusFour := "available: 1(1)"
   115  	wgBCWait.Done()
   116  	wgBCExit.Wait()
   117  	actual = mo.String()
   118  	if actual != expectedStatusFour {
   119  		t.Logf("Status Four: expected: %s actual: %s", expectedStatusFour, actual)
   120  		t.Fail()
   121  	}
   122  
   123  	seq := Append("")
   124  	// expected sequence: a¢dAcCbB
   125  	// ¢d and cCbB may flip
   126  	if len(seq) != len(expectedSequence) ||
   127  		seq[0:1] != expectedSequence[0:1] ||
   128  		seq[exitAIndex:exitAIndex+1] != expectedSequence[exitAIndex:exitAIndex+1] {
   129  		t.Logf("Bad sequence: expected: %s actual: %s", expectedSequence, seq)
   130  		t.Fail()
   131  	}
   132  }
