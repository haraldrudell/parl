     8  import "time"

    10  // Alert impelemnts a renewable alert timer
    11  type Alert struct {
    12  	time.Duration
    13  	*time.Timer
    14  	Fired bool
    15  }
    16  
    17  // NewAlert allocates a renewable alert time
    18  func NewAlert(d time.Duration) (al *Alert) {
    19  	al = &Alert{Duration: d}
    20  	return
    21  }
    22  
    23  // Start initializes a new alert period
    24  func (al *Alert) Start() {
    25  	if al.Timer != nil && !al.Fired {
    26  		al.Timer.Stop()
    27  	}
    28  	al.Timer = time.NewTimer(al.Duration)
    29  	al.Fired = false
    30  }
    31  
    32  // Stop releases resources associated with this alert
    33  func (al *Alert) Stop() {
    34  	if al.Timer != nil && !al.Fired {
    35  		al.Timer.Stop()
    36  	}
    37  	al.Timer = nil
    38  }
