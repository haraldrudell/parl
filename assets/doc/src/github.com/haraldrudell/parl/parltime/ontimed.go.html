    15  // OnTimed provides events to perform on the hour
    16  type OnTimed struct {
    17  	nowChannel chan time.Time
    18  	sdChan     chan struct{}
    19  	isClose    parl.AtomicBool
    20  	ctx        context.Context
    21  }
    22  
    23  func GetTimer(period time.Duration, timeZone Tz, ctx context.Context) (ot *OnTimed) {
    24  	if ctx == nil {
    25  		ctx = context.Background()
    26  	}
    27  	ot = &OnTimed{nowChannel: make(chan time.Time), sdChan: make(chan struct{}), ctx: ctx}
    28  	if period == 0 {
    29  		return
    30  	}
    31  	go ot.run(period, timeZone)
    32  	return
    33  }
    34  
    35  // NewOnTimedLocal provides events to perform on the hour
    36  func NewOnTimedLocal(period time.Duration) (ot *OnTimed) {
    37  	return GetTimer(period, LOCAL, nil)
    38  }
    39  
    40  // NewOnTimed provides events to perform on the hour
    41  func NewOnTimed(period time.Duration) (ot *OnTimed) {
    42  	return GetTimer(period, UTC, nil)
    43  }
    44  
    45  // NewOnTimed initializes OnTimed
    46  func (ot *OnTimed) run(period time.Duration, timeZone Tz) {
    47  	defer parl.Recover("OnTimed.run", nil, func(err error) { parl.Info("%+v\n", err) })
    48  	var timer *time.Timer
    49  	defer func() {
    50  		if timer != nil {
    51  			timer.Stop()
    52  		}
    53  	}()
    54  	for {
    55  		timer = OnTimer(period, timeZone)
    56  		select {
    57  		case <-ot.ctx.Done():
    58  		case <-ot.sdChan:
    59  		case now := <-timer.C:
    60  			ot.nowChannel <- now
    61  			continue
    62  		}
    63  		break
    64  	}
    65  }
    66  
    67  // Chan gets a channel to wait for
    68  func (ot *OnTimed) Chan() (ch <-chan time.Time) {
    69  	return ot.nowChannel
    70  }
    71  
    72  // Close stop the durations
    73  func (ot *OnTimed) Close() {
    74  	if ot.isClose.Set() {
    75  		close(ot.sdChan)
    76  	}
    77  }
