     8  import (
     9  	"context"
    10  	"time"
    11  )
    12  
    13  const (
    14  	TimerCh TriggeringChan = iota
    15  	DoneCh
    16  	ValueCh
    17  )
    18  
    19  type TriggeringChan uint8
    20  
    21  // Value is an event value that is being debounced
    22  type Value interface{}
    23  
    24  // SenderFunc takes an untyped value, type asserts and sends on a typed channel
    25  type SenderFunc func([]Value)
    26  
    27  // ReceiverFunc takes two channels, listens to them and a typed channel, returns what channel triggered and a possible untyped value
    28  type ReceiverFunc func(c <-chan time.Time, done <-chan struct{}) (which TriggeringChan, value Value)
    29  
    30  // Debouncer debounces event streams of Value
    31  func NewDebouncer(d time.Duration, receiver ReceiverFunc, sender SenderFunc, ctx context.Context) (err error) {
    32  	var values []Value
    33  	var timer *time.Timer
    34  	c := make(<-chan time.Time)
    35  	C := c
    36  	defer func() {
    37  		if timer != nil {
    38  			timer.Stop()
    39  		}
    40  	}()
    41  	for {
    42  		switch which, value := receiver(C, ctx.Done()); which {
    43  		case ValueCh:
    44  			if timer == nil {
    45  				timer = time.NewTimer(d)
    46  				C = timer.C
    47  			}
    48  			values = append(values, value)
    49  			continue
    50  		case TimerCh:
    51  			timer = nil
    52  			C = c
    53  			sender(values)
    54  			values = nil
    55  			continue
    56  		case DoneCh:
    57  		}
    58  		break
    59  	}
    60  	return
    61  }
