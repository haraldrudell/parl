   134  func applyYaml(bytes []byte, unmarshal mains.UnmarshalFunc, yamlDictionaryKey string) (hasData bool, err error) {
   135  	parl.Debug("applyYaml: bytes: %d key: %q\nbytes: %q\n", len(bytes), yamlDictionaryKey, string(bytes))
   136  
   137  	yamlContentObject := map[string]*YamlData{}                 // need map[string] because yaml top level is dictionary
   138  	if err = unmarshal(bytes, &yamlContentObject); err != nil { // populate mapp
   139  		err = parl.Errorf("applyYaml unmarshal: '%w'", err)
   140  		return
   141  	}
   142  	parl.Debug("applyYaml: object: %+v\n", yamlContentObject) // map[options:0xc0000723c0]
   143  	yamlDataPointer := yamlContentObject[yamlDictionaryKey]   // pick out the options dictionary value
   144  	if hasData = yamlDataPointer != nil; !hasData {
   145  		parl.Debug("applyYaml: yaml key had no data")
   146  		return
   147  	}
   148  	y = *yamlDataPointer
   149  	parl.Debug("applyYaml: data: %+v\n", y)
   150  	return
   151  }
