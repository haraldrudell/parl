     8  import "sync/atomic"
     9  
    10  type AtomicBool struct {
    11  	value int32 // atomic
    12  }
    13  
    14  const (
    15  	abTrue  = int32(1)
    16  	abFalse = int32(0)
    17  )
    18  
    19  func (ab *AtomicBool) IsTrue() (isTrue bool) {
    20  	return atomic.LoadInt32(&ab.value) == abTrue
    21  }
    22  
    23  func (ab *AtomicBool) Set() (wasNotSet bool) {
    24  	return atomic.SwapInt32(&ab.value, abTrue) != abTrue
    25  }
    26  
    27  func (ab *AtomicBool) Clear() (wasSet bool) {
    28  	return atomic.SwapInt32(&ab.value, abFalse) == abTrue
    29  }
