     8  import (
     9  	"sync"
    10  )
    11  
    12  type FanOut struct {
    13  	wg      sync.WaitGroup
    14  	ErrCh   chan error
    15  	Results chan interface{}
    16  }
    17  
    18  type FanThunk func() (result interface{}, err error)
    19  type FanProc func() (err error)
    20  
    21  func NewFanOut() (fo *FanOut) {
    22  	return &FanOut{
    23  		ErrCh:   make(chan error),
    24  		Results: make(chan interface{}),
    25  	}
    26  }
    27  
    28  // Do executes a procedure in a goroutine that has no result other than a possible non-nil error
    29  func (cr *FanOut) Do(name string, proc FanProc) {
    30  	cr.wg.Add(1)
    31  	go cr.fanout(name, nil, proc)
    32  }
    33  
    34  // Run executes a thunk in a goroutine with a possible non-nil result and a possible non-nil error
    35  func (cr *FanOut) Run(name string, thunk FanThunk) {
    36  	cr.wg.Add(1)
    37  	go cr.fanout(name, thunk, nil)
    38  }
    39  
    40  func (cr *FanOut) fanout(name string, thunk FanThunk, proc FanProc) {
    41  	defer cr.wg.Done()
    42  	var err error
    43  	defer Recover(name, &err, func(e error) { cr.ErrCh <- err })
    44  
    45  	if thunk == nil {
    46  		err = proc()
    47  	} else {
    48  		var result interface{}
    49  		result, err = thunk()
    50  		if result != nil {
    51  			cr.Results <- result
    52  		}
    53  	}
    54  }
    55  
    56  // Wait waits for all Do and Run invocations to complete, then shuts down
    57  func (cr *FanOut) Wait() {
    58  	cr.wg.Wait()
    59  	close(cr.ErrCh)
    60  	close(cr.Results)
    61  }
