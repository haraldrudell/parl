    60  func writeBytes(filename string, bytes []byte) (err error) {
    61  	var file *os.File
    62  	if file, err = os.OpenFile(filename, os.O_WRONLY|os.O_TRUNC|os.O_CREATE, 0600); err != nil {
    63  		err = error116.Errorf("os.OpenFile %q: '%w'", filename, err)
    64  		return
    65  	}
    66  	defer func() {
    67  		if e := file.Close(); e != nil {
    68  			err = error116.AppendError(err, e)
    69  		}
    70  	}()
    71  	_, err = file.Write(bytes)
    72  	return
    73  }
