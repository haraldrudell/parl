    18  var defaultTime, _ = time.Parse(time.RFC3339, "2000-01-01T00:00:00Z")

    20  // ContextTime obtains a shared timestamp
    21  func ContextTime(ctx context.Context) (t time.Time, found bool) {
    22  	var timePointer *time.Time
    23  	contextValue := ctx.Value(TimePointer)
    24  	if timePointer, found = contextValue.(*time.Time); found {
    25  		t = *timePointer
    26  	} else {
    27  		t = defaultTime
    28  	}
    29  	return
    30  }
