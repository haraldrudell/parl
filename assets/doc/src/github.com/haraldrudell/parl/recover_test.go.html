    12  func TestResultPanic(t *testing.T) {
    13  
    14  	// test recover from panic
    15  	doFn("PANIC", // printed test name prefix
    16  		"panic", // string used for panic
    17  		"Recover from panic in parl.parlPanic: 'non-error value: string panic'", // expected error text
    18  		parlPanic, // the function to invoke
    19  		t)         // test for printing

    44  func doFn(testID, text, expectedErr string, fn func(text string, fne func(error)) (err error), t *testing.T) {
    45  	// variables
    46  	var errValue error
    47  	var actualErr error
    48  	//var actualInt int
    49  	var actualBool bool
    50  	//var actual string
    51  	fne := func(e error) {
    52  		actualBool = true
    53  		actualErr = e
    54  	}
    55  
    56  	// invoke test function
    57  
    58  	errValue = fn(text, fne)
    59  
    60  	if expectedErr == "" {
    61  		if actualBool {
    62  			t.Logf("%s fne was invoked", testID)
    63  			t.Fail()
    64  		}
    65  		if errValue != nil {
    66  			t.Logf("%s returned an error: %v", testID, errValue)
    67  			t.Fail()
    68  		}
    69  		if actualErr != nil {
    70  			t.Logf("%s fne received an error: %v", testID, actualErr)
    71  			t.FailNow()
    72  		}
    73  	} else {
    74  		if !actualBool {
    75  			t.Logf("%s fne was not invoked", testID)
    76  			t.Fail()
    77  		}
    78  		if errValue == nil {
    79  			t.Logf("%s function did not return error", testID)
    80  			t.Fail()
    81  		} else {
    82  			if errValue.Error() != expectedErr {
    83  				t.Logf("%s returned error: expected: %q actual: %s", testID, expectedErr, errValue)
    84  				t.Fail()
    85  			}
    86  		}
    87  		if actualErr == nil {
    88  			t.Logf("%s function did not return error", testID)
    89  			t.FailNow()
    90  		} else {
    91  			if actualErr.Error() != expectedErr {
    92  				t.Logf("%s fne recived error: expected: %q actual: %q", testID, actualErr, actualErr.Error())
    93  				t.Fail()
    94  			}
    95  		}
    96  	}
    97  }
