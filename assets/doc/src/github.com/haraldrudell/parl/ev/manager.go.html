    20  // ThreadManager provides thread-safe management of go routines
    21  type ThreadManager struct {
    22  	ctx     context.Context    // context with cancel
    23  	cancel  context.CancelFunc // cancel for exactly this context
    24  	events  chan Event         // channel for goroutine output consumed by caller
    25  	goIDMap sync.Map           //[GoID]bool thread-safe
    26  	count   int64              // atomic. Because ranging goIDMap needs lock, store count outside
    27  	Launch  time.Time
    28  }

    34  // NewManager obtains instance for executing goroutines
    35  func NewManager(ctx0 context.Context) (mgr Manager) {
    36  	if ctx0 == nil {
    37  		ctx0 = context.Background()
    38  	}
    39  	ctx, cancel := context.WithCancel(ctx0)
    40  	return &ThreadManager{ctx: ctx, cancel: cancel, events: make(chan Event)}
    41  }

    48  // CalleeContext produces an environment for calling a goroutine - thread safe
    49  func (mgr *ThreadManager) CalleeContext(ID ...string) (env Callee) {
    50  	threadNo := atomic.AddInt64(&nextName, 1)
    51  	var name string
    52  	if len(ID) > 0 {
    53  		name = ID[0]
    54  	}
    55  	if name == "" {
    56  		name = fmt.Sprintf("thread-%d", threadNo)
    57  	}
    58  	thread := NewEvThread(name)
    59  	env = NewCallee(name, thread.ID, mgr.events, mgr.ctx)
    60  	mgr.goIDMap.Store(thread.ID, thread)
    61  	atomic.AddInt64(&mgr.count, 1)
    62  	return
    63  }

    65  // ProcessExit indicates whether this event is the final event from a terminating goroutine
    66  func (mgr *ThreadManager) ProcessEvent(ev Event) (err error) {
    67  	if ev == nil {
    68  		return error116.New("goroutine channel closed or goroutine sent nil")
    69  	}
    70  	gID := ev.GoID()
    71  	if _, ok := mgr.goIDMap.Load(gID); !ok {
    72  		return error116.New("event from unknown goroutine")
    73  	}
    74  	if _, ok := ev.(*ExitEvent); ok {
    75  		mgr.goIDMap.Delete(gID)
    76  		atomic.AddInt64(&mgr.count, -1) // decrement
    77  	}
    78  	return
    79  }

    81  // Action determines whether the executable should continue to wait for additional threads
    82  func (mgr *ThreadManager) Action(threadResult error, action CancelAction) (isEnd bool) {
    83  	// a goroutine terminated with threadResult
    84  	if action == TerminateAll || // on first terminating thread, all threads are terminated
    85  		(action == WhileOk && threadResult != nil) { // go until error, and this is failing thread
    86  		return mgr.Cancel()
    87  	}
    88  	// state: KeepGoing or (WhileOk and no error)
    89  	if threadResult != nil { // ev.Keepgoing and errors
    90  		strs := []string{threadResult.Error()}
    91  		if errorList, ok := threadResult.(error116.ErrorHasList); ok {
    92  			for _, e := range errorList.ErrorList() {
    93  				strs = append(strs, e.Error())
    94  			}
    95  		}
    96  		parl.Info("goroutine errors: %v", strings.Join(strs, "\n"))
    97  	}
    98  	return mgr.IsEnd()
    99  }

   112  func (mgr *ThreadManager) Threads() (names []string, IDs []GoID) {
   113  	mgr.goIDMap.Range(func(key, value interface{}) bool {
   114  		thread := getMapValue(value)
   115  		names = append(names, thread.Name)
   116  		IDs = append(IDs, thread.ID)
   117  		return true
   118  	})
   119  	return
   120  }

   122  func getMapValue(value interface{}) (thread *EvThread) {
   123  	var ok bool
   124  	if thread, ok = value.(*EvThread); !ok {
   125  		panic(parl.Errorf("Bad value type in thread map: %T", value))
   126  	}
   127  	return
   128  }
