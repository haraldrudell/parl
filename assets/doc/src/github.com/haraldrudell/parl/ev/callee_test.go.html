    39  func doFn(testID, expectedErr string, fn func(ctx Callee) (err error), t *testing.T) {
    40  	// values
    41  	name := "name"
    42  
    43  	// context
    44  	gID := uuid.New()
    45  	eventCh := make(chan Event, 3)
    46  	ctx := NewCallee(name, GoID(gID), eventCh, nil)
    47  
    48  	// variables
    49  	var errValue error
    50  	var event Event
    51  	var actualInt int
    52  	var actualBool bool
    53  	var actual string
    54  
    55  	// invoke test function
    56  	errValue = fn(ctx)
    57  
    58  	if expectedErr == "" {
    59  		if errValue != nil {
    60  			t.Logf("%s ctx.Result set err: %v", testID, errValue)
    61  			t.Fail()
    62  		}
    63  	} else {
    64  		if errValue == nil {
    65  			t.Logf("%s ctx.Result did not set err", testID)
    66  			t.FailNow()
    67  		}
    68  		actual = errValue.Error()
    69  		if actual != expectedErr {
    70  			t.Logf("%s ctx.Result err: expected: %q actual: %s", testID, expectedErr, errValue)
    71  			t.FailNow()
    72  		}
    73  	}
    74  	actualInt = len(eventCh)
    75  	if actualInt != 1 {
    76  		t.Logf("%s ctx.Result did not send 1 message on panic: %d", testID, actualInt)
    77  		t.FailNow()
    78  	}
    79  	if event, actualBool = <-eventCh; !actualBool {
    80  		t.Logf("%s Read event channel failed", testID)
    81  		t.FailNow()
    82  	}
    83  	switch e := event.(type) {
    84  	case *ExitEvent:
    85  		if e.Err != errValue {
    86  			t.Logf("%s ExitEvent bad Err: expected: %v actual: %v", testID, errValue, e.Err)
    87  			t.Fail()
    88  		}
    89  	default:
    90  		t.Logf("%s Unknown event type: %T", testID, event)
    91  		t.FailNow()
    92  	}
    93  }

    95  func TestSend(t *testing.T) {
    96  	value := 3
    97  	type DataType int
    98  	var event Event
    99  
   100  	mgr := NewManager(nil)
   101  	closer := make(chan struct{})
   102  
   103  	// the goroutine
   104  	go func(ctx Callee) {
   105  		defer close(closer)
   106  		defer ctx.Result(nil)
   107  		ctx.Send(DataType(value))
   108  	}(mgr.CalleeContext())
   109  
   110  	// get the listen event containing the address
   111  	getEvent := func() (event Event, isEnd bool) {
   112  		select {
   113  		case <-closer:
   114  			isEnd = true
   115  			return
   116  		case event = <-mgr.Events():
   117  		}
   118  		t.Logf("Event: %T %[1]v", event)
   119  		if err := mgr.ProcessEvent(event); err != nil {
   120  			t.Errorf("Bad event: %T %[1]v", event)
   121  			t.FailNow()
   122  		}
   123  		if exitEv, ok := event.(*ExitEvent); ok {
   124  			if exitEv.Err != nil {
   125  				t.Errorf("Failure: %T %[1]v", event)
   126  				t.FailNow()
   127  			}
   128  		}
   129  		if dataEv, ok := event.(*DataEvent); ok {
   130  			if warn, ok := dataEv.Payload().(evx.Warning); ok {
   131  				t.Errorf("Warning: %T %[1]v", warn)
   132  				t.FailNow()
   133  			}
   134  		}
   135  		return
   136  	}
   137  
   138  	event, _ = getEvent()
   139  	if dataEvent, ok := event.(*DataEvent); !ok {
   140  		t.Errorf("Received other than DataEvent: %T %[1]v", event)
   141  		return
   142  	} else if data, ok := dataEvent.Payload().(DataType); !ok {
   143  		t.Errorf("Payload not ListenEvent: %T", dataEvent.Payload())
   144  		return
   145  	} else {
   146  		t.Logf("Value: %T %[1]v", data)
   147  	}
   148  }
