     6  /*
     7  Package ev provides standardized goroutine management
     8  
     9  events contain thread completions, failures and any type of data items. A manager launches and controls a set of goroutines
    10  
    11  ctx.Result is defered by a gorutine, captures panics and sends the goroutine’s result using Success or Failure
    12  
    13    func (ctx ev.Callee) {
    14      var err error
    15      defer func() {
    16        ctx.Result(&err, recover())
    17      }()
    18    …
    19  
    20  © 2020–present Harald Rudell <harald.rudell@gmail.com> (https://haraldrudell.github.io/haraldrudell/)
    21  */
    22  package ev
    23  
    24  import (
    25  	"context"
    26  
    27  	"github.com/google/uuid"
    28  )
    29  
    30  // EventRx receives events from goroutines
    31  type EventRx <-chan Event
    32  
    33  // GoID is a unique value identifying a goroutine
    34  type GoID uuid.UUID
    35  
    36  // Callee provides functions for a called goroutine
    37  type Callee interface {
    38  	context.Context                                // Deadline Done Err Value
    39  	Success()                                      // Success indicates this goruote terminated successfully
    40  	Failure(err error)                             // Failure indicates this goroutine failed with err
    41  	Result(errp *error)                            // recovery for gorutines: usage: defer ctx.Result(&err)
    42  	ResultV(errp *error, recoverValue interface{}) // recovery for gorutines
    43  	Thread() (name string, ID GoID)                // Thread provides information about the thread
    44  	Send(payload interface{})                      // Send allows a goroutine to send custom types
    45  }
    46  
    47  const (
    48  	// TerminateAll terminates other goroutines on first termination
    49  	TerminateAll CancelAction = iota
    50  	// KeepGoing waits for all goroutines to complete even on errors
    51  	KeepGoing
    52  	// WhileOk terminates if any goroutine errors
    53  	WhileOk
    54  )
    55  
    56  // CancelAction holds strategy for when a goruotine terminates
    57  type CancelAction uint8
    58  
    59  // Event is a message passed from goroutines
    60  type Event interface {
    61  	GoID() (gID GoID) // [16]byte
    62  }
    63  
    64  // Manager provides management of goroutines
    65  type Manager interface {
    66  	Cancel() (isEnd bool)                    // Cancel signals to all goroutines to terminate
    67  	Events() (ch EventRx)                    // Events provides the channel emitting events from goroutines
    68  	CalleeContext(ID ...string) (ctx Callee) // CalleeContext context for a new goroutine, 0 or 1 argument
    69  	ProcessEvent(event Event) (err error)    // ProcessEvent sanity checks an EventRx event
    70  	Action(evResult error, action CancelAction) (awaitThreads bool)
    71  	IsEnd() (isEnd bool)                   // IsEnd returns true if all goroutines have terminated
    72  	Threads() (names []string, IDs []GoID) // Threads provide information on currently managed goroutines
    73  }
