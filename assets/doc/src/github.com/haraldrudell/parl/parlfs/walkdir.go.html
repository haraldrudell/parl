    14  // Directory is a file system entry with children
    15  type Directory struct {
    16  	Entry
    17  	Children []FSEntry
    18  }
    19  
    20  // NewDirectory instantiates FSEntry that can have children
    21  func NewDirectory(path string, entry *Entry, sym func(string)) FSEntry {
    22  	dir := Directory{Entry: *entry}
    23  	var entryNames []string
    24  	if entryNames, dir.Err = readDir(path); dir.Err != nil {
    25  		return &dir
    26  	}
    27  	dir.Children = make([]FSEntry, len(entryNames))
    28  	for i, entryName := range entryNames {
    29  		dir.Children[i] = NewEntry(filepath.Join(path, entryName), entryName, sym)
    30  	}
    31  	return &dir
    32  }

    34  // Walk traverses a built root
    35  func (dir *Directory) Walk(path string, walkFunc filepath.WalkFunc) error {
    36  	if err := walkFunc(path, dir.FileInfo, dir.Err); err != nil {
    37  		return err
    38  	}
    39  	for _, entry := range dir.Children {
    40  		if err := entry.Walk(filepath.Join(path, entry.SafeName()), walkFunc); err != nil && err != filepath.SkipDir {
    41  			return err
    42  		}
    43  	}
    44  	return nil
    45  }

    47  // readDir reads the directory named by dirname and returns
    48  // a sorted list of directory entries.
    49  func readDir(path string) ([]string, error) {
    50  	f, err := os.Open(path)
    51  	if err != nil {
    52  		return nil, err
    53  	}
    54  	names, err := f.Readdirnames(-1)
    55  	f.Close()
    56  	if err != nil {
    57  		return nil, err
    58  	}
    59  	sort.Strings(names)
    60  	return names, nil
    61  }
