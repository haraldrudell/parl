     8  import (
     9  	"os"
    10  	"path/filepath"
    11  )
    12  
    13  // Entry is a file system entry that can be part of directory, symlink or other
    14  type Entry struct {
    15  	Base        string // we can have entries with FileInfo nil, so base path must be stored
    16  	os.FileInfo        // info.Name() has basename, os.FileInfo: interface
    17  	Err         error  // a symlink may fail to resolve
    18  }
    19  
    20  // NewEntry returns Entry or Directory
    21  func NewEntry(path string, base string, sym func(string)) FSEntry {
    22  	en := Entry{Base: base}
    23  	en.FileInfo, en.Err = os.Lstat(path)
    24  	if en.Err != nil {
    25  		return &en
    26  	}
    27  	if en.Mode()&os.ModeSymlink != 0 {
    28  		var symlinkTarget string
    29  		if symlinkTarget, en.Err = filepath.EvalSymlinks(path); en.Err != nil {
    30  			return &en
    31  		}
    32  		sym(symlinkTarget) // possibly add new root or restructure due to link to parent
    33  	}
    34  	if en.IsDir() {
    35  		return NewDirectory(path, &en, sym)
    36  	}
    37  	return &en
    38  }
    39  
    40  // SafeName gets filepath.Base() in a safe way
    41  func (en *Entry) SafeName() string {
    42  	if en.FileInfo != nil {
    43  		return en.FileInfo.Name()
    44  	}
    45  	return en.Base
    46  }
    47  
    48  // Walk traverses a built root
    49  func (en *Entry) Walk(path string, walkFunc filepath.WalkFunc) error {
    50  	return walkFunc(path, en.FileInfo, en.Err)
    51  }
