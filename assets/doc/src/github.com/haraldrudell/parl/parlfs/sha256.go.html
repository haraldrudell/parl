    22  // Sha256 contains sha-256 hash
    23  type Sha256 []byte
    24  
    25  // Sha256Context get sha256 of a file with context
    26  func Sha256Context(ctx context.Context, filename string) (s2 Sha256, err error) {
    27  	var file *os.File
    28  	if file, err = os.Open(filename); err != nil {
    29  		return
    30  	}
    31  	defer func() {
    32  		if e := file.Close(); e != nil {
    33  			err = error116.AppendError(err, e)
    34  		}
    35  	}()
    36  	hash := sha256.New()
    37  	if _, err = io.Copy(hash, NewContextReader(ctx, file)); err != nil {
    38  		return
    39  	}
    40  	s2 = hash.Sum(nil)
    41  	return
    42  }
    43  
    44  // Valid determines if hash is present
    45  func (s2 *Sha256) Valid() bool {
    46  	return s2 != nil && len(*s2) == byteLength
    47  }
    48  
    49  func (s2 Sha256) String() string {
    50  	return hex.EncodeToString(s2)
    51  }
