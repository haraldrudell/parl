    14  // FSEntry represnt a branch in a file system hierarchy
    15  // FSEntry is implemented by Entry (file) or Directory or Root (walk entry point)
    16  type FSEntry interface {
    17  	SafeName() string
    18  	IsDir() bool
    19  	Walk(path string, walkFunc filepath.WalkFunc) error
    20  }
    21  
    22  // Root is a file system hierarchy
    23  type Root struct {
    24  	Path    string // path as provided
    25  	Evaled  string // path after filepath.EvalSymlinks
    26  	Err     error
    27  	FSEntry // Directory or Entry
    28  }
    29  
    30  // NewRoot allocates a root
    31  func NewRoot(path string) (rt *Root, err error) {
    32  	root := Root{Path: path}
    33  	if root.Evaled, err = AbsEval(path); err != nil {
    34  		return
    35  	}
    36  	rt = &root
    37  	return
    38  }
    39  
    40  // Build scans the file system for this root
    41  func (rt *Root) Build(sym func(string)) {
    42  	if rt.FSEntry != nil {
    43  		panic(parl.New("root.Build invoked more than once"))
    44  	}
    45  	rt.FSEntry = NewEntry(rt.Path, "", sym)
    46  }
    47  
    48  // Walk traverses the root
    49  func (rt *Root) Walk(walkFunc filepath.WalkFunc) error {
    50  	if rt.FSEntry == nil {
    51  		panic(parl.New("root.Walk without root.Build"))
    52  	}
    53  
    54  	// the root node is special, because the entire path is already known
    55  	return rt.FSEntry.Walk(rt.Path, walkFunc)
    56  }
