    18  // AbsEval makes path absolute and resolves symlinks
    19  func AbsEval(path string) (p string, err error) {
    20  	if p, err = filepath.Abs(path); err == nil {
    21  		p, err = filepath.EvalSymlinks(p)
    22  	}
    23  	return
    24  }

    26  // Dirs retrieves absolute paths to all directories, while following symlinks, from initial dir argument
    27  func Dirs(dir string) (dirs []string, err error) {
    28  
    29  	// make dir path absolute
    30  	var dir0 string
    31  	if dir0, err = filepath.Abs(dir); err != nil {
    32  		return
    33  	}
    34  
    35  	// find directories to watch
    36  	err = Walk(dir0, func(path string, info os.FileInfo, err0 error) (err error) {
    37  		if err0 != nil {
    38  			return err0
    39  		}
    40  		if info.IsDir() {
    41  			dirs = append(dirs, path)
    42  		}
    43  		return
    44  	})
    45  	return
    46  }
