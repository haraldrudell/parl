     8  // errorList implements a list of errors that belong together,
     9  // as opposed to an error chain intended for being built at
    10  // different levels of the code.
    11  // an error list is commonly built using error116.AppendError(err, err2) error
    12  type errorList struct {
    13  	ErrorChain         // errorList implements error chain, ie. rich data associated with a single error
    14  	errs       []error // errorList has a list of additional errors
    15  }
    16  
    17  var _ error = &errorList{}        // errorList implements the error116.Wrapper interface, ie. is an error chain
    18  var _ ErrorHasList = &errorList{} // errorList implements the error116.ErrorHasList and error interfaces
    19  
    20  func (et *errorList) Append(e2 error) (e error) {
    21  	if et == nil {
    22  		panic(New("errorList.Append on zero value"))
    23  	}
    24  	if e2 == nil {
    25  		return et // noop return
    26  	}
    27  	et.errs = append(et.errs, e2)
    28  	return et
    29  }
    30  
    31  func (et *errorList) ErrorList() (errs []error) {
    32  	if et != nil && len(et.errs) > 0 {
    33  		errs = make([]error, len(et.errs))
    34  		copy(errs, et.errs)
    35  	}
    36  	return
    37  }
    38  
    39  func (et *errorList) Count() int {
    40  	if et == nil {
    41  		return 0
    42  	}
    43  	return len(et.errs)
    44  }
