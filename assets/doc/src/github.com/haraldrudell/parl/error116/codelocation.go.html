    15  // CodeLocation is similar to runtime.Frame, but includes basic types
    16  // string and int only
    17  type CodeLocation struct { //
    18  	File string // /opt/foxyboy/sw/privates/parl/mains/executable.go
    19  	Line int
    20  	// package path ending in package name, optional type information, function name
    21  	// github.com/haraldrudell/parl/mains.(*Executable).AddErr
    22  	FuncName string
    23  }

    25  func NewCodeLocation(stackFramesToSkip int) (cl *CodeLocation) {
    26  	if stackFramesToSkip < 0 {
    27  		stackFramesToSkip = 0
    28  	}
    29  	c := CodeLocation{}
    30  
    31  	var pc uintptr
    32  	var ok bool
    33  	// pc: opaque
    34  	// file: basename.go
    35  	// line: int 25
    36  	if pc, c.File, c.Line, ok = runtime.Caller(1 + stackFramesToSkip); !ok {
    37  		panic(New("runtime.Caller failed"))
    38  	}
    39  	// rFunc: runtime.Func is all opaque fields. methods:
    40  	// Entry() (uintptr)
    41  	// FileLine(uintptr) (line string, line int) "/opt/foxyboy/sw/privates/parl/mains/executable.go"
    42  	// Name(): github.com/haraldrudell/parl/mains.(*Executable).AddErr
    43  	if rFunc := runtime.FuncForPC(pc); rFunc != nil {
    44  		c.FuncName = rFunc.Name()
    45  	}
    46  	return &c
    47  }

    49  func GetCodeLocation(rFrame *runtime.Frame) (cl *CodeLocation) {
    50  	// runtime.Frame:
    51  	// PC uintptr, Func *Func, Function string, File string, Line int, Entry uintptr, funcInfo funcInfo
    52  	return &CodeLocation{
    53  		File:     rFrame.File,
    54  		Line:     rFrame.Line,
    55  		FuncName: rFrame.Function,
    56  	}
    57  }

    64  func (cl *CodeLocation) Name() (funcName string) {
    65  	packageAndFunc := filepath.Base(cl.FuncName)
    66  	if lastDotIndex := strings.LastIndex(packageAndFunc, "."); lastDotIndex >= 0 {
    67  		return packageAndFunc[lastDotIndex+1:]
    68  	}
    69  	return packageAndFunc
    70  }

    73  func (cl *CodeLocation) Package() (funcName string) {
    74  	packageAndFunc := filepath.Base(cl.FuncName)
    75  	if dotIndex := strings.Index(packageAndFunc, "."); dotIndex >= 0 {
    76  		return packageAndFunc[:dotIndex]
    77  	}
    78  	return packageAndFunc
