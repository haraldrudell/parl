    66  // HasStack detects if the error chain already contains a stack trace
    67  func HasStack(err error) (hasStack bool) {
    68  	var e ErrorCallStacker
    69  	return errors.As(err, &e)
    70  }

    72  // GetStackTrace gets the last stack trace
    73  func GetStackTrace(err error) (stack StackSlice) {
    74  	var e ErrorCallStacker
    75  	if errors.As(err, &e) {
    76  		stack = e.StackTrace()
    77  	}
    78  	return
    79  }

    81  // GetInnerMostStack gets the oldest stack trace in the error chain
    82  func GetInnerMostStack(err error) (stack StackSlice) {
    83  	var e ErrorCallStacker
    84  	for errors.As(err, &e) {
    85  	}
    86  	if e != nil {
    87  		stack = e.StackTrace()
    88  	}
    89  	return
    90  }

    92  // DumpChain gets a space-separated string of error implementation type names
    93  func DumpChain(err error) (s string) {
    94  	var strs []string
    95  	for err != nil {
    96  		strs = append(strs, fmt.Sprintf("%T", err))
    97  		err = errors.Unwrap(err)
    98  	}
    99  	s = strings.Join(strs, "\x20")
   100  	return
   101  }
