    28  func (e *errorStack) ChainString(format Format) (s string) {
    29  	if e == nil {
    30  		return
    31  	} else if format == DefaultFormat {
    32  		s = e.Error()
    33  		return
    34  	}
    35  	if format != ShortSuffix && format != LongSuffix {
    36  		s = e.Error()
    37  	}
    38  	if format == LongFormat || format == LongSuffix {
    39  		s += e.StackSlice.String()
    40  	} else {
    41  		s += e.StackSlice.Short()
    42  	}
    43  	var chainString ChainStringer
    44  	errors.As(e.error, &chainString)
    45  	if chainString == nil {
    46  		return
    47  	}
    48  	if format == LongFormat {
    49  		format = LongSuffix
    50  	} else if format == ShortFormat {
    51  		format = ShortSuffix
    52  	}
    53  	s += chainString.ChainString(format)
    54  	return
    55  }

    57  func (e *errorStack) Format(s fmt.State, verb rune) {
    58  	switch verb {
    59  	case 'v':
    60  		if format := PrintfFormat(s); format != DefaultFormat {
    61  			io.WriteString(s, e.ChainString(format))
    62  			return
    63  		}
    64  		fallthrough // %v is same as %s
    65  	case 's':
    66  		io.WriteString(s, e.Error())
    67  	case 'q':
    68  		fmt.Fprintf(s, "%q", e.Error())
    69  	}
    70  }
