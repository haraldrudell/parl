    10  // AllErrors obtains all error objects of all error lists that err and its error chain contains.
    11  // If err is nil, errs is empty otherwise errs[0] == err.
    12  // If err and its error chain has no lists, errs contain only err
    13  func AllErrors(err error) (errs []error) {
    14  	if err != nil {
    15  		errs = append(errs, err)
    16  	}
    17  
    18  	// traverse all found errors
    19  	errMap := map[error]bool{}
    20  	for errsIndex := 0; errsIndex < len(errs); errsIndex++ {
    21  
    22  		// avoid cyclic error graph
    23  		e := errs[errsIndex]
    24  		_, knownError := errMap[e]
    25  		if knownError {
    26  			continue
    27  		}
    28  		errMap[e] = true
    29  
    30  		// traverse the error chain except the first entry which is the error itself
    31  		errorChain := ErrorChainSlice(e)
    32  		for ecIndex := 1; ecIndex < len(errorChain); ecIndex++ {
    33  
    34  			// include a possible error list
    35  			errs = append(errs, ErrorList(errorChain[ecIndex])...)
    36  		}
    37  	}
    38  	return
    39  }
    40  
    41  // ErrorChainSlice returns a slice of errors from a possible error chain.
    42  // If err is nil, an empty slice is returned.
    43  // If err does not have an error chain, a slice of only err is returned.
    44  // Otherwise, the slice lists each error in the chain starting with err at index 0.
    45  func ErrorChainSlice(err error) (errs []error) {
    46  	for err != nil {
    47  		errs = append(errs, err)
    48  		err = errors.Unwrap(err)
    49  	}
    50  	return
    51  }
    52  
    53  // ErrorList obtains the error list associated with err.
    54  // is err has no list associated or the list is empty, len(errs) == 0
    55  // err itself is not returned
    56  func ErrorList(err error) (errs []error) {
    57  	if errorListInstance, ok := err.(*errorList); ok {
    58  		errs = errorListInstance.ErrorList()
    59  	}
    60  	return
    61  }

    63  func AppendError(err error, e2 error) (e error) {
    64  	if e2 == nil {
    65  		return err // noop return
    66  	}
    67  	if err == nil {
    68  		return e2 // single error return
    69  	}
    70  
    71  	// check if err has list already
    72  	if eList, ok := err.(ErrorHasList); ok {
    73  		return eList.Append(e2)
    74  	}
    75  	return &errorList{ErrorChain{err}, []error{e2}}
    76  }
