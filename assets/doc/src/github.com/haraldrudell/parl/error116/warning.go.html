    12  // WarningType is an error with lesser impact
    13  type WarningType struct {
    14  	ErrorChain
    15  }

    19  // Warning indicates a problem of less severity than error
    20  func Warning(err error) error {
    21  	return Stack(&WarningType{ErrorChain{err}})
    22  }
    23  
    24  // IsWarning determines if an error is a warning
    25  func IsWarning(err error) bool {
    26  	var warning WarningType
    27  	return errors.Is(err, warning)
    28  }

    30  func (w WarningType) Error() (s string) {
    31  	if err := w.Unwrap(); err != nil {
    32  		return err.Error()
    33  	}
    34  	return "warning"
    35  }
