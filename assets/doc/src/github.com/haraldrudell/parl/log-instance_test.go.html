   139  func TestRegexpLogI(t *testing.T) {
   140  
   141  	// What is matched against the regexp is FuncName:
   142  	// github.com/haraldrudell/parl.TestRegexpLogI
   143  	// fully qualified package name and function name
   144  	if false {
   145  		t.Logf("FuncName: %q", error116.NewCodeLocation(0).FuncName)
   146  		t.FailNow()
   147  	}
   148  
   149  	text1, textNewline, expectedLocation, regexpLocation, writer, lg := mocksLogI()
   150  	var actualSlice []string
   151  	var actual string
   152  
   153  	nonMatchingRegexp := "aaa"
   154  	matchingRegexp := regexpLocation
   155  
   156  	// string that regExp is matched against: "github.com/haraldrudell/parl.TestRegexpLogI"
   157  	//t.Logf("string that regExp is matched against: %q", error116.NewCodeLocation(0).FuncName)
   158  
   159  	// matching regexp
   160  	if err := lg.SetRegexp(matchingRegexp); err != nil {
   161  		t.Logf("SetRegexp failed: input: %q err: %+v", matchingRegexp, err)
   162  		t.Fail()
   163  	}
   164  	lg.Debug(textNewline)
   165  	actualSlice = writer.getData()
   166  	if len(actualSlice) != 1 {
   167  		t.Logf("matching regexp did not print 1: %d regexp input:\n%q compiled:\n%+v",
   168  			len(actualSlice),
   169  			matchingRegexp,
   170  			lg.infoRegexp)
   171  		t.Fail()
   172  	}
   173  	actual = actualSlice[0]
   174  	if !strings.HasPrefix(actual, text1) {
   175  		t.Logf("matching regexp: missing prefix: %q text: %q", text1, actual)
   176  		t.Fail()
   177  	}
   178  	if !strings.Contains(actual, expectedLocation) {
   179  		t.Logf("matching regexp: no location: actual:\n%q expected:\n%q",
   180  			actual,
   181  			expectedLocation)
   182  		t.Fail()
   183  	}
   184  	if strings.Index(actual, "\n") != len(actual)-1 {
   185  		t.Logf("matching regexp: newline not at end: actual: %q expected: %q", actual, regexpLocation)
   186  		t.Fail()
   187  	}
   188  
   189  	// non-matching regexp
   190  	if err := lg.SetRegexp(nonMatchingRegexp); err != nil {
   191  		panic(err)
   192  	}
   193  	lg.Debug(text1)
   194  	actualSlice = writer.getData()
   195  	if len(actualSlice) > 0 {
   196  		t.Logf("non-matching regexp did print: %d", len(actualSlice))
   197  		t.Fail()
   198  	}
   199  }

   222  func quoteSliceLogI(sList []string) (s string) {
   223  	var s2 []string
   224  	for _, sx := range sList {
   225  		s2 = append(s2, fmt.Sprintf("%q", sx))

   230  func mocksLogI() (text1, textNewline, expectedLocation, regexpLocation string, writer *mockWriterLogI, lg *LogInstance) {
   231  	text1 = "abc"
   232  	textNewline = text1 + "\n"
   233  
   234  	// location text for this file
