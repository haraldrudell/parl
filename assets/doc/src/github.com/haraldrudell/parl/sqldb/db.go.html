     6  // Package sqldb interfaces database/sql
     7  package sqldb
     8  
     9  import (
    10  	"context"
    11  	"database/sql"
    12  
    13  	"github.com/haraldrudell/parl"
    14  )
    15  
    16  const (
    17  	contextDBformat = "db.db"
    18  )
    19  
    20  // StoreDB saves db handle in context
    21  func StoreDB(ctx context.Context, db *sql.DB) {
    22  	parl.StoreInContext(ctx, contextDBformat, db)
    23  }
    24  
    25  // GetDB obtains db handle from context
    26  func GetDB(ctx context.Context) (db *sql.DB) {
    27  	db, ok := ctx.Value(contextDBformat).(*sql.DB)
    28  	if !ok {
    29  		panic(parl.New("No db in context"))
    30  	}
    31  	return
    32  }
    33  
    34  // DiscardDB removes db handle from context
    35  func DiscardDB(ctx context.Context) {
    36  	parl.DelFromContext(ctx, contextDBformat)
    37  }
