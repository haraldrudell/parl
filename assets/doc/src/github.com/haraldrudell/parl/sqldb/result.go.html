     6  package sqldb

     8  import (
     9  	"database/sql"
    10  	"strconv"
    11  )
    12  
    13  // Result makes sql.Result printable
    14  type Result struct {
    15  	sql.Result
    16  }
    17  
    18  // Get obtains last id and number of affected rows
    19  func (r *Result) Get() (id int64, count int64, sErr string, err error) {
    20  	id, err = r.LastInsertId()
    21  	if err == nil {
    22  		count, err = r.RowsAffected()
    23  		if err != nil {
    24  			sErr = "sql.Result.RowsAffected error: " + err.Error()
    25  		}
    26  	} else {
    27  		sErr = "sql.Result.LastInsertId error: " + err.Error()
    28  	}
    29  	return
    30  }
    31  
    32  func (r Result) String() (s string) {
    33  	id, count, sErr, _ := r.Get()
    34  
    35  	s = "sql.Result: "
    36  	if sErr != "" {
    37  		s += sErr
    38  	} else {
    39  		s += "last inserted id: " + strconv.FormatInt(id, 10) +
    40  			"rows affected: " + strconv.FormatInt(count, 10)
    41  	}
    42  	return
    43  }
