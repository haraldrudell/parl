    33  func Invocation(stackFramesToSkip int) (s string) {
    34  	if stackFramesToSkip < 0 {
    35  		stackFramesToSkip = 0
    36  	}
    37  	linesToSkip := (stackFramesToSkip + skipFrames) * linesPerStackFrame
    38  	remainingStackTrace := string(debug.Stack())
    39  	var goRoutineText string
    40  	if index := strings.Index(remainingStackTrace, "\n"); index >= 0 {
    41  		goRoutineText = remainingStackTrace[:index] // "goroutine 1 [running]:"
    42  		remainingStackTrace = remainingStackTrace[index+1:]
    43  	}
    44  	for linesToSkip > 0 {
    45  		if index := strings.Index(remainingStackTrace, "\n"); index >= 0 {
    46  			remainingStackTrace = remainingStackTrace[index+1:]
    47  		}
    48  		linesToSkip--
    49  	}
    50  	return goRoutineText + "\n" + remainingStackTrace
    51  }

    53  func GoRoutineID() (ID string) {
    54  	stackTraceS := string(debug.Stack())
    55  	var firstLine string
    56  	if index := strings.Index(stackTraceS, "\n"); index >= 0 { // first line: "goroutine 1 [running]:"
    57  		firstLine = stackTraceS[:index]
    58  	}
    59  	firstLine = strings.TrimPrefix(firstLine, goroutinePrefix) // leading "goroutine "
    60  	firstLine = strings.TrimSuffix(firstLine, goroutineSuffix) // trailing colon
    61  	if strings.HasSuffix(firstLine, goroutineEndBracket) {     // remove possible ending bracketed text "[…]"
    62  		if index := strings.LastIndex(firstLine, goroutineStartBracket); index >= 0 {
    63  			firstLine = firstLine[:index]
    64  		}
    65  	}
    66  	return strings.TrimSuffix(firstLine, goRoutineSpaceSuffix) // ending space
    67  }
