    16  	defaultParallelism = 20
    17  	uint64MinusOne     = ^uint64(0)

    20  // Moderator invokes functions at a limited level of parallelism
    21  type Moderator struct {
    22  	parallelism  uint64
    23  	shutdownLock sync.Once
    24  	isShutdown   AtomicBool
    25  	dataLock     sync.Mutex
    26  	available    uint64 // behind dataLock
    27  	waitCount    uint64 // atomic
    28  	waitLock     sync.Mutex
    29  }
    30  
    31  var ErrModeratorShutdown = errors.New("Moderator shut down")
    32  
    33  // NewModerator creates a new Moderator used to limit parallelism
    34  func NewModerator(parallelism uint64) (mo *Moderator) {
    35  	if parallelism == 0 {
    36  		parallelism = defaultParallelism
    37  	}
    38  	return &Moderator{parallelism: parallelism, available: parallelism}
    39  }
    40  
    41  // Do calls fn limited by the moderator’s parallelism.
    42  // If the moderator is shut down, ErrModeratorShutdown is returned
    43  func (mo *Moderator) Do(fn func() error) (err error) {
    44  	if fn == nil {
    45  		panic(New("Moderator.Do with nil function"))
    46  	}
    47  	if mo.isShutdown.IsTrue() {
    48  		return ErrModeratorShutdown
    49  	}
    50  	defer mo.returnTicket() // we will always get a ticket, and it should be returned
    51  	if err = mo.getTicket(); err != nil {
    52  		return // shutdown
    53  	}
    54  	return fn()
    55  }
    56  
    57  func (mo *Moderator) getTicket() (err error) {
    58  	if mo.getAvailableTicket() {
    59  		return // a ticket was available
    60  	}
    61  
    62  	// wait for ticket
    63  	mo.waitLock.Lock()
    64  
    65  	// we now have a ticket!
    66  	if mo.isShutdown.IsTrue() {
    67  		return ErrModeratorShutdown
    68  	}
    69  	return
    70  }
    71  
    72  func (mo *Moderator) getAvailableTicket() (hasTicket bool) {
    73  	mo.dataLock.Lock()
    74  	defer mo.dataLock.Unlock()
    75  	if mo.available > 0 {
    76  		mo.available--
    77  		if mo.available == 0 {
    78  			mo.waitLock.Lock() // the next thread will wait
    79  		}
    80  		return true
    81  	}
    82  	mo.waitCount++
    83  	return
    84  }
    85  
    86  func (mo *Moderator) returnTicket() {
    87  	mo.dataLock.Lock()
    88  	defer mo.dataLock.Unlock()
    89  
    90  	// hand ticket to the queue
    91  	if mo.waitCount > 0 {
    92  		mo.waitCount--
    93  		mo.waitLock.Unlock()
    94  		return
    95  	}
    96  	if mo.available == 0 {
    97  		mo.waitLock.Unlock()
    98  	}
    99  
   100  	// note an additional ticket available
   101  	mo.available++
   102  }
   103  
   104  func (mo *Moderator) Status() (parallelism uint64, available uint64, waiting uint64, isShutdown bool) {
   105  	parallelism = mo.parallelism
   106  	mo.dataLock.Lock()
   107  	available = mo.available
   108  	mo.dataLock.Unlock()
   109  	waiting = atomic.LoadUint64(&mo.waitCount)
   110  	isShutdown = mo.isShutdown.IsTrue()
   111  	return
   112  }
   113  
   114  func (mo *Moderator) String() (s string) {
   115  	p, a, w, sd := mo.Status()
   116  	if a > 0 {
   117  		s = fmt.Sprintf("available: %d(%d)", a, p)
   118  	} else {
   119  		s = fmt.Sprintf("waiting: %d(%d)", w, p)
   120  	}
   121  	if sd {
   122  		s += " shutdown"
   123  	}
   124  	return
   125  }

   127  func (mo *Moderator) shutdown() {
   128  	mo.isShutdown.Set()
   129  }
   130  
   131  func (mo *Moderator) Shutdown() {
   132  	mo.shutdownLock.Do(mo.shutdown)
   133  }
